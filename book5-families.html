Family Concepts

ClassDescriptionAnalogy
FamilyThe family container (.rfa)A class definition
FamilySymbolA specific type/size within the familyA class with specific values
FamilyInstanceA placed instance in the modelAn object/instance of the class
C#
// Navigate the Family → Symbol → Instance hierarchy
FamilyInstance fi = ...;
FamilySymbol   sym    = fi.Symbol;
Family         family = sym.Family;

string familyName = family.Name;
string typeName   = sym.Name;

// Get all types in a family
foreach (ElementId symId in family.GetFamilySymbolIds())
{
    FamilySymbol s = doc.GetElement(symId) as FamilySymbol;
}

// Get all family instances of a category
var doors = new FilteredElementCollector(doc)
    .OfCategory(BuiltInCategory.OST_Doors)
    .OfClass(typeof(FamilyInstance))
    .Cast<FamilyInstance>()
    .ToList();
// 02

Load & Place Families

C#
using (var tx = new Transaction(doc, "Load Family"))
{
    tx.Start();

    // Load a .rfa family into the project
    Family family;
    bool loaded = doc.LoadFamily(
        @"C:\Families\MyDoor.rfa", out family);

    if (loaded)
    {
        // Get the first symbol from the loaded family
        FamilySymbol sym = doc.GetElement(
            family.GetFamilySymbolIds().First()) as FamilySymbol;

        if (!sym.IsActive) sym.Activate();   // REQUIRED before placing!

        // Place on a wall host
        FamilyInstance fi = doc.Create.NewFamilyInstance(
            new XYZ(5, 0, 0),
            sym,
            wall,
            StructuralType.NonStructural);

        // Place free-standing (no host)
        FamilyInstance chair = doc.Create.NewFamilyInstance(
            new XYZ(3, 3, 0),
            chairSym,
            level,
            StructuralType.NonStructural);
    }

    tx.Commit();
}

// Load with duplication handling option
doc.LoadFamily(path, new FamilyLoadOptions(), out var f2);

// Implement IFamilyLoadOptions to control overwrite behavior
public class FamilyLoadOptions : IFamilyLoadOptions
{
    public bool OnFamilyFound(bool familyInUse, out bool overwriteParamValues)
    { overwriteParamValues = true; return true; }
    public bool OnSharedFamilyFound(Family sf, bool familyInUse,
        out FamilySource src, out bool overwriteParams)
    { src = FamilySource.Family; overwriteParams = true; return true; }
}
// 03

Family Documents

Open a family document programmatically to modify or create types.

C#
// Edit an existing family
Document famDoc = doc.EditFamily(family);

// Or open a .rfa directly
Document famDoc2 = app.Application.OpenDocumentFile(@"C:\Fam.rfa");

if (famDoc.IsFamilyDocument)
{
    using (var tx = new Transaction(famDoc, "Edit Family"))
    {
        tx.Start();
        // ... edit family content ...
        tx.Commit();
    }

    // Load back into the host project
    famDoc.LoadFamily(doc, new FamilyLoadOptions());
    famDoc.Close(false);    // false = don't save separately
}
// 04

Family Parameters

C#
// FamilyManager manages types and parameters
FamilyManager mgr = famDoc.FamilyManager;

using (var tx = new Transaction(famDoc, "Family Params"))
{
    tx.Start();

    // Add a new parameter
    FamilyParameter param = mgr.AddParameter(
        "MyWidth",                          // name
        GroupTypeId.Dimensions,             // group
        SpecTypeId.Length,                  // type
        true);                               // isInstance

    // Set value for current type
    mgr.Set(param, UnitUtils.ConvertToInternalUnits(900, UnitTypeId.Millimeters));

    // Add a new family type
    mgr.NewType("1200 Wide");
    mgr.Set(param, UnitUtils.ConvertToInternalUnits(1200, UnitTypeId.Millimeters));

    // Iterate all types
    foreach (FamilyType ft in mgr.Types)
    {
        string typeName = ft.Name;
        mgr.CurrentType = ft;
        double val = mgr.GetValue(param).AsDouble();
    }

    tx.Commit();
}
// 05

Family Geometry — Extrusions & Blends

C#
// Create an extrusion in a family document
SketchPlane sketchPlane = SketchPlane.Create(famDoc,
    Plane.CreateByNormalAndOrigin(XYZ.BasisZ, XYZ.Zero));

// Build a closed CurveArrArray profile
var loop = new CurveArray();
loop.Append(Line.CreateBound(new XYZ(0,0,0), new XYZ(1,0,0)));
loop.Append(Line.CreateBound(new XYZ(1,0,0), new XYZ(1,1,0)));
loop.Append(Line.CreateBound(new XYZ(1,1,0), new XYZ(0,1,0)));
loop.Append(Line.CreateBound(new XYZ(0,1,0), new XYZ(0,0,0)));

var profile = new CurveArrArray();
profile.Append(loop);

using (var tx = new Transaction(famDoc, "Create Extrusion"))
{
    tx.Start();

    Extrusion extrusion = famDoc.FamilyCreate.NewExtrusion(
        true,         // isSolid
        profile,
        sketchPlane,
        2.0);         // height in feet

    // Assign a material
    ElementId matId = new FilteredElementCollector(famDoc)
        .OfClass(typeof(Material)).Cast<Material>()
        .First(m => m.Name.Contains("Concrete")).Id;

    extrusion.get_Parameter(BuiltInParameter.MATERIAL_ID_PARAM)
        .Set(matId);

    tx.Commit();
}

// 07

Materials

C#
// Get all materials
var materials = new FilteredElementCollector(doc)
    .OfClass(typeof(Material))
    .Cast<Material>()
    .ToList();

// Read material properties
Material mat = materials.First();
string  name       = mat.Name;
Color   surfaceColor = mat.Color;
Color   cutColor     = mat.CutForegroundPatternColor;
int     transparency = mat.Transparency;

// Create a new material
using (var tx = new Transaction(doc, "Create Material"))
{
    tx.Start();

    ElementId newMatId = Material.Create(doc, "My Red Material");
    Material  newMat   = doc.GetElement(newMatId) as Material;

    newMat.Color        = new Color(200, 50, 50);
    newMat.Transparency = 20;

    tx.Commit();
}

// Apply material to a wall
using (var tx = new Transaction(doc, "Apply Material"))
{
    tx.Start();
    Parameter matParam = wall.get_Parameter(
        BuiltInParameter.MATERIAL_ID_PARAM);
    if (matParam != null) matParam.Set(newMatId);
    tx.Commit();
}
