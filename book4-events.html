// 01

Events Overview

Revit exposes events at three levels. Subscribe in OnStartup(), unsubscribe in OnShutdown().

LevelObjectExamples
Applicationapp.ControlledApplicationDocumentOpened, DocumentSaving, DocumentCreated, DocumentClosed
UIuiapp (UIControlledApplication)ViewActivated, SelectionChanged, Idling, ApplicationInitialized
Elementapp.ControlledApplicationDocumentChanged (read-only), use DMU for write reactions
// Golden Rule You CANNOT call Revit API from outside threads (background tasks, modeless dialogs, timers). The only safe bridges are: Idling event and ExternalEvent + IExternalEventHandler. Use ExternalEvent — it's more efficient.
// 02

Application Events

C# — IExternalApplication
public class MyApp : IExternalApplication
{
    public Result OnStartup(UIControlledApplication app)
    {
        var cApp = app.ControlledApplication;

        // Document lifecycle events
        cApp.DocumentOpened   += OnDocumentOpened;
        cApp.DocumentCreated  += OnDocumentCreated;
        cApp.DocumentClosing  += OnDocumentClosing;
        cApp.DocumentSaving   += OnDocumentSaving;
        cApp.DocumentSaved    += OnDocumentSaved;

        // Changed event (read-only, fires after every tx commit/undo/redo)
        cApp.DocumentChanged  += OnDocumentChanged;

        // UI events
        app.ViewActivated     += OnViewActivated;
        app.Idling            += OnIdling;
        app.ApplicationInitialized += OnAppInit;

        return Result.Succeeded;
    }

    public Result OnShutdown(UIControlledApplication app)
    {
        // Always unsubscribe to prevent memory leaks!
        app.ControlledApplication.DocumentOpened  -= OnDocumentOpened;
        app.ControlledApplication.DocumentSaving  -= OnDocumentSaving;
        app.ControlledApplication.DocumentChanged -= OnDocumentChanged;
        app.ViewActivated                         -= OnViewActivated;
        app.Idling                                -= OnIdling;
        return Result.Succeeded;
    }
}

UI Events

private void OnAppInit(object sender, ApplicationInitializedEventArgs e)
{
    // Safe place to do one-time startup work

        sender as Autodesk.Revit.ApplicationServices.Application);

private void OnIdling(object sender, IdlingEventArgs e)
{
    // Fires during idle time — valid API context
    // WARNING: will hog CPU if you don't manage frequency!

    // Set frequency to reduce CPU load
    e.SetRaiseWithoutDelay(false);  // slower, kinder to CPU

    UIApplication uiapp = sender as UIApplication;
    if (uiapp?.ActiveUIDocument == null) return;
    // ... do periodic work ...
}

External Events — The Bridge Pattern

External Events are the correct way to call Revit API from a modeless window, background task, or any non-API thread. Idling works but consumes CPU — use ExternalEvent instead.

ApproachWhen to UseCPU Impact
ExternalEvent + IExternalEventHandlerModeless dialogs, async workflows, WPF formsLow — fires only when Raise() called
Idling EventPeriodic polling, simple animationsHigh — fires constantly
Direct callFrom IExternalCommand.Execute() onlyN/A — synchronous
// How it works 1. Create a class implementing IExternalEventHandler. 2. Register it with ExternalEvent.Create(handler) in OnStartup or Execute. 3. From your WPF button: call externalEvent.Raise(). 4. Revit calls handler.Execute(uiapp) at the next idle cycle in a valid API context.
// 06

IExternalEventHandler — Full Implementation

C# — Step 1: The Handler
// Step 1: Implement the handler
public class MyEventHandler : IExternalEventHandler
{
    // Thread-safe data passing via public properties
    public string ParameterName  { get; set; }
    public string ParameterValue { get; set; }
    public bool   Success        { get; private set; }

    // Execute is called by Revit in a valid API context
    public void Execute(UIApplication uiapp)
    {
        try
        {
            UIDocument uidoc = uiapp.ActiveUIDocument;
            if (uidoc == null) { Success = false; return; }

            Document doc = uidoc.Document;

            using (var tx = new Transaction(doc, "External Event Action"))
            {
                tx.Start();

                // Use the data passed via properties
                var walls = new FilteredElementCollector(doc)
                    .OfClass(typeof(Wall)).Cast<Wall>().ToList();

                foreach (var wall in walls)
                {
                    var p = wall.LookupParameter(ParameterName);
                    if (p != null && !p.IsReadOnly)
                        p.Set(ParameterValue);
                }

                tx.Commit();
                Success = true;
            }
        }
        catch (Exception ex)
        {
            Success = false;
            TaskDialog.Show("Error", ex.Message);
        }
    }

    public string GetName() => "My External Event Handler";
}
C# — Step 2: Register in OnStartup
public class MyApplication : IExternalApplication
{
    // Static so WPF window can access it
    public static ExternalEvent   MyExternalEvent  { get; private set; }
    public static MyEventHandler  MyHandler        { get; private set; }

    public Result OnStartup(UIControlledApplication app)
    {
        // Create handler and register it
        MyHandler       = new MyEventHandler();
        MyExternalEvent = ExternalEvent.Create(MyHandler);

        return Result.Succeeded;
    }

    public Result OnShutdown(UIControlledApplication app)
    {
        MyExternalEvent.Dispose();
        return Result.Succeeded;
    }
}
C# — Step 3: Raise from WPF button
// Inside your WPF window button click handler:
private void BtnApply_Click(object sender, RoutedEventArgs e)
{
    // Pass data to the handler via properties
    MyApplication.MyHandler.ParameterName  = txtParamName.Text;
    MyApplication.MyHandler.ParameterValue = txtValue.Text;

    // Raise the event — Revit will call Execute() at next idle cycle
    ExternalEventRequest req = MyApplication.MyExternalEvent.Raise();

    // Check that the request was accepted
    if (req == ExternalEventRequest.Accepted)
        lblStatus.Content = "Request queued...";
    else
        lblStatus.Content = $"Request failed: {req}";
}

ExternalEventRequest Status Values

StatusMeaning
AcceptedRevit accepted and queued the request ✅
PendingA previous raise is still pending — wait
DeniedRevit is busy (modal dialog open, not in idle state)
TimedOutRevit did not process the event in time

Modeless Dialog — Complete Pattern

The full pattern for a WPF modeless window that interacts with Revit.

C# — Full modeless pattern
// The modeless WPF window
public partial class MyModelessWindow : Window
{
    private readonly ExternalEvent  _event;
    private readonly MyEventHandler _handler;

    public MyModelessWindow(
        ExternalEvent ev, MyEventHandler handler)
    {
        InitializeComponent();
        _event   = ev;
        _handler = handler;
    }

    private void BtnRename_Click(object sender, RoutedEventArgs e)
    {
        _handler.ParameterName  = "Mark";
        _handler.ParameterValue = txtNewValue.Text;

        var result = _event.Raise();

        lblStatus.Content = result == ExternalEventRequest.Accepted
            ? "Queued ✔"
            : $"Failed: {result}";
    }

    // Important: don't close, just hide when user clicks X
    protected override void OnClosing(CancelEventArgs e)
    {
        e.Cancel = true;
        Hide();
    }
}

// Launch from an IExternalCommand
[Transaction(TransactionMode.Manual)]
public class ShowWindowCommand : IExternalCommand
{
    private static MyModelessWindow _window;

    public Result Execute(ExternalCommandData cd,
        ref string msg, ElementSet els)
    {
        if (_window == null || !_window.IsLoaded)
        {
            _window = new MyModelessWindow(
                MyApplication.MyExternalEvent,
                MyApplication.MyHandler);
        }
        _window.Show();
        return Result.Succeeded;
    }
}

Dynamic Model Update (DMU)

DMU lets you modify the model in response to changes — unlike DocumentChanged which is read-only. Implement IUpdater and register triggers.

C# — IUpdater Implementation
public class WallTagUpdater : IUpdater
{
    private static AddInId   _appId;
    private static UpdaterId _updaterId;

    public WallTagUpdater(AddInId appId)
    {
        _appId     = appId;
        _updaterId = new UpdaterId(appId,
            new Guid("12345678-1234-1234-1234-123456789abc"));
    }

    // Called when a watched element changes
    public void Execute(UpdaterData data)
    {
        Document doc = data.GetDocument();

        foreach (ElementId id in data.GetModifiedElementIds())
        {
            Wall wall = doc.GetElement(id) as Wall;
            if (wall == null) continue;

            // Example: auto-update wall mark when wall moves
            Parameter mark = wall.get_Parameter(BuiltInParameter.ALL_MODEL_MARK);
            if (mark != null && !mark.IsReadOnly)
                mark.Set($"W-{wall.Id.IntegerValue}");
        }
    }

    public string    GetUpdaterName()    => "Wall Tag Updater";
    public string    GetAdditionalInformation() => "Auto-tags walls on change";
    public ChangePriority GetChangePriority() => ChangePriority.MEPFixtures;
    public UpdaterId GetUpdaterId()      => _updaterId;
}

// Register the updater in OnStartup:
public Result OnStartup(UIControlledApplication app)
{
    var updater = new WallTagUpdater(app.ActiveAddInId);
    UpdaterRegistry.RegisterUpdater(updater);

    // Add trigger: fire on geometry change of any wall
    ElementClassFilter wallFilter = new ElementClassFilter(typeof(Wall));
    UpdaterRegistry.AddTrigger(
        updater.GetUpdaterId(),
        wallFilter,
        Element.GetChangeTypeGeometry());

    return Result.Succeeded;
}

// Unregister in OnShutdown:
public Result OnShutdown(UIControlledApplication app)
{
    UpdaterRegistry.UnregisterUpdater(_updater.GetUpdaterId());
    return Result.Succeeded;
}

Change Types for DMU Triggers

Change TypeFires When
Element.GetChangeTypeGeometry()Element geometry changes (move, resize)
Element.GetChangeTypeParameter(param)A specific parameter value changes
Element.GetChangeTypeAny()Any change to the element
Element.GetChangeTypeElementAddition()Element is added to the document
Element.GetChangeTypeElementDeletion()Element is deleted

Failure Handling

Implement IFailuresPreprocessor to suppress or handle Revit warnings and errors that occur during a transaction.

C#
// Suppress all warnings silently
public class SuppressWarnings : IFailuresPreprocessor
{
    public FailureProcessingResult PreprocessFailures(
        FailuresAccessor accessor)
    {
        var failures = accessor.GetFailureMessages();

        foreach (var failure in failures)
        {
            if (failure.GetSeverity() == FailureSeverity.Warning)
                accessor.DeleteWarning(failure);   // suppress warnings
            else
                return FailureProcessingResult.ProceedWithRollBack; // errors → rollback
        }

        return FailureProcessingResult.Continue;
    }
}

// Use it in a transaction:
using (var tx = new Transaction(doc, "Safe Create"))
{
    FailureHandlingOptions opts = tx.GetFailureHandlingOptions();
    opts.SetFailuresPreprocessor(new SuppressWarnings());
    tx.SetFailureHandlingOptions(opts);

    tx.Start();
    // ... work that might cause warnings ...
    tx.Commit();
}

Idling Event

The Idling event fires between user interactions. Use it for periodic tasks but prefer ExternalEvent for modeless dialogs.

C#
private int _idleCount = 0;

private void OnIdling(object sender, IdlingEventArgs e)
{
    UIApplication uiapp = sender as UIApplication;
    if (uiapp?.ActiveUIDocument == null) return;

    _idleCount++;

    // Only run every 100 idle cycles to save performance
    if (_idleCount % 100 != 0) return;

    // Valid API context — can make changes
    Document doc = uiapp.ActiveUIDocument.Document;

    // Tell Revit not to call this as fast as possible
    e.SetRaiseWithoutDelay(false);

    // Use ExternalEvent instead of Idling when possible!
}
// Recommendation Prefer ExternalEvent over Idling for modeless dialog interactions. Idling fires constantly and degrades performance. ExternalEvent only fires when you explicitly call Raise(), giving Revit full control.
