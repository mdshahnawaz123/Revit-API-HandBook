& MEP Connectors

// 01

MEP Overview

ClassCategorySystem

// 02

Pipes & Plumbing

C#
// Get pipe type and system type
PipeType pipeType = new FilteredElementCollector(doc)
    .OfClass(typeof(PipeType)).Cast<PipeType>().First();

PipingSystemType sysType = new FilteredElementCollector(doc)
    .OfClass(typeof(PipingSystemType))
    .Cast<PipingSystemType>().First();

using (var tx = new Transaction(doc, "Create Pipe"))
{
    tx.Start();

    // Create a straight pipe segment
    Pipe pipe = Pipe.Create(
        doc,
        sysType.Id,          // system type
        pipeType.Id,         // pipe type
        level.Id,            // reference level
        new XYZ(0,0,3),     // start point
        new XYZ(5,0,3));    // end point

    // Set diameter
    pipe.get_Parameter(BuiltInParameter.RBS_PIPE_DIAMETER_PARAM)
        .Set(UnitUtils.ConvertToInternalUnits(100, UnitTypeId.Millimeters));

    tx.Commit();
}

// Read pipe properties
Pipe p = ...;
double diameter  = p.get_Parameter(BuiltInParameter.RBS_PIPE_DIAMETER_PARAM).AsDouble();
double length    = p.get_Parameter(BuiltInParameter.CURVE_ELEM_LENGTH).AsDouble();
XYZ    direction = (p.Location as LocationCurve).Curve.GetEndPoint(1)
                 - (p.Location as LocationCurve).Curve.GetEndPoint(0);

Ducts & Mechanical

C#
DuctType ductType = new FilteredElementCollector(doc)
    .OfClass(typeof(DuctType)).Cast<DuctType>().First();

MechanicalSystemType mSysType = new FilteredElementCollector(doc)
    .OfClass(typeof(MechanicalSystemType))
    .Cast<MechanicalSystemType>().First();

using (var tx = new Transaction(doc, "Create Duct"))
{
    tx.Start();

    Duct duct = Duct.Create(
        doc,
        mSysType.Id,
        ductType.Id,
        level.Id,
        new XYZ(0,0,3),
        new XYZ(5,0,3));

    // Set width and height (rectangular duct)
    duct.get_Parameter(BuiltInParameter.RBS_CURVE_WIDTH_PARAM)
        .Set(UnitUtils.ConvertToInternalUnits(400, UnitTypeId.Millimeters));
    duct.get_Parameter(BuiltInParameter.RBS_CURVE_HEIGHT_PARAM)
        .Set(UnitUtils.ConvertToInternalUnits(200, UnitTypeId.Millimeters));

    tx.Commit();
}

Connectors

Connectors are the endpoints of MEP elements. Connect two elements by linking their connectors.

C#
// Get connectors from a pipe
ConnectorManager cm   = pipe.ConnectorManager;
ConnectorSet     cons = cm.Connectors;

Connector start = null, end = null;
foreach (Connector c in cons)
{
    if (c.ConnectorType == ConnectorType.End)
    {
        if (start == null) start = c;
        else               end   = c;
    }
}

// Check connector properties
XYZ    origin   = start.Origin;
double diameter = start.Radius * 2;
bool   isConnected = start.IsConnected;

// Connect two pipes end-to-end
using (var tx = new Transaction(doc, "Connect Pipes"))
{
    tx.Start();

    Connector c1 = GetConnectorAt(pipe1, pipe2.GetEndPoint(0));
    Connector c2 = GetConnectorAt(pipe2, pipe1.GetEndPoint(1));

    if (c1 != null && c2 != null)
        c1.ConnectTo(c2);

    tx.Commit();
}

// Helper: find closest connector to a point
Connector GetConnectorAt(Element elem, XYZ pt, double tol = 0.001)
{
    foreach (Connector c in elem.GetMEPModel().ConnectorManager.Connectors)
        if (c.Origin.DistanceTo(pt) < tol) return c;
    return null;
}

Rooms & Spaces

Spaces are MEP-specific versions of Rooms — they carry load calculations and HVAC data.

C#
// Get MEP Spaces (different from architectural Rooms!)
var spaces = new FilteredElementCollector(doc)
    .OfCategory(BuiltInCategory.OST_MEPSpaces)
    .Cast<Space>()
    .ToList();

foreach (var space in spaces)
{
    string name   = space.Name;
    double area   = space.Area;          // sq ft
    double volume = space.Volume;        // cubic ft

    // HVAC loads
    double coolingLoad = space.get_Parameter(
        BuiltInParameter.ROOM_ACTUAL_COOLING_LOAD_PARAM)
        ?.AsDouble() ?? 0;

    // Check space/room relationship
    Room linkedRoom = space.Room;
    if (linkedRoom != null)
    {
        string roomName = linkedRoom.Name;
    }
}

// Create a Space
using (var tx = new Transaction(doc, "Create Space"))
{
    tx.Start();
    Space newSpace = doc.Create.NewSpace(
        level,
        new UV(5, 5));    // location as UV on the level
    newSpace.Name = "Office 101";
    tx.Commit();
}
