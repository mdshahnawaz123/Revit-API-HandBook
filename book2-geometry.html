// 01

XYZ, Vectors & Transform

XYZ is used for both points and vectors. Transform handles rotation, translation and scaling.

C#
// Creating points and vectors
XYZ origin  = XYZ.Zero;                         // (0,0,0)
XYZ pt      = new XYZ(3.0, 0.0, 0.0);          // in feet!
XYZ up      = XYZ.BasisZ;                        // (0,0,1)

// Vector operations
XYZ    a      = new XYZ(1,0,0);
XYZ    b      = new XYZ(0,1,0);
double dot    = a.DotProduct(b);                // scalar product
XYZ    cross  = a.CrossProduct(b);              // perpendicular
XYZ    norm   = a.Normalize();                   // unit vector
double len    = a.GetLength();
double dist   = a.DistanceTo(b);

// Transform (position + orientation)
Transform t = Transform.Identity;
t.Origin     = new XYZ(5, 0, 0);
t.BasisX     = XYZ.BasisX;
t.BasisY     = XYZ.BasisY;
t.BasisZ     = XYZ.BasisZ;

// Create rotation transform
Transform rot = Transform.CreateRotation(
    XYZ.BasisZ, Math.PI / 2);          // 90° around Z

// Line creation
Line line  = Line.CreateBound(pt, pt + new XYZ(10,0,0));
// 02

Walls

There are four wall kinds: Basic, Stacked, Curtain, Unknown. The API works best with Basic walls.

WallKindDescriptionAPI Support
BasicStandard single-layer or compound wallFull — create, modify, read
StackedMultiple walls stacked verticallyLimited — cannot get sub-walls via API
CurtainGlass curtain wall systemLimited — panels/mullions separate
UnknownUnresolved typeRead only
C#
// Create a wall
Line     line     = Line.CreateBound(new XYZ(0,0,0), new XYZ(10,0,0));
Level    level    = GetLevel(doc, "Level 1");
WallType wallType = new FilteredElementCollector(doc)
    .OfClass(typeof(WallType)).Cast<WallType>().First();

using (var tx = new Transaction(doc, "Create Wall"))
{
    tx.Start();
    Wall wall = Wall.Create(doc, line, wallType.Id, level.Id,
        3.0,    // height in feet
        0.0,    // offset from level
        false, // flip
        false); // structural
    tx.Commit();
}

// Read wall properties
Wall w = ...;
WallKind     kind     = w.WallType.Kind;
bool         flipped  = w.Flipped;
double       width    = w.Width;                // in feet
LocationCurve loc     = w.Location as LocationCurve;
Curve        curve    = loc.Curve;
XYZ          start    = curve.GetEndPoint(0);
XYZ          end      = curve.GetEndPoint(1);
double       length   = curve.Length;           // in feet

// Flip wall orientation
using (var tx = new Transaction(doc, "Flip"))
{
    tx.Start();
    w.flip();   // flips exterior/interior face
    tx.Commit();
}

// Wall Location Line values (WALL_KEY_REF_PARAM)
// 0=Centerline, 1=Core Centerline
// 2=Finish Face:Exterior, 3=Finish Face:Interior
// 4=Core Face:Exterior, 5=Core Face:Interior
Parameter locLine = w.get_Parameter(BuiltInParameter.WALL_KEY_REF_PARAM);
locLine.Set(2); // Finish Face Exterior
// Wall.Flipped The Wall.Flipped property and Wall.flip() method control which face is the exterior. Orientation is (0,1,0) before flip and (0,-1,0) after. Location line value 3 = Finish Face Interior means the wall moved but location not changed.

Floors & Ceilings

C#
// Create a floor from a profile
FloorType floorType = new FilteredElementCollector(doc)
    .OfClass(typeof(FloorType)).Cast<FloorType>().First();

// Build a closed loop (CurveLoop)
var profile = new CurveLoop();
profile.Append(Line.CreateBound(new XYZ(0,0,0), new XYZ(10,0,0)));
profile.Append(Line.CreateBound(new XYZ(10,0,0), new XYZ(10,10,0)));
profile.Append(Line.CreateBound(new XYZ(10,10,0), new XYZ(0,10,0)));
profile.Append(Line.CreateBound(new XYZ(0,10,0), new XYZ(0,0,0)));

using (var tx = new Transaction(doc, "Create Floor"))
{
    tx.Start();
    Floor floor = Floor.Create(doc,
        new List<CurveLoop> { profile },
        floorType.Id,
        level.Id);
    tx.Commit();
}

Roofs

Roofs are created using FootPrintRoof (most common), ExtrusionRoof, or via RoofBase.

C#
RoofType roofType = new FilteredElementCollector(doc)
    .OfClass(typeof(RoofType)).Cast<RoofType>().First();

ModelCurveArray footprint = new ModelCurveArray();

using (var tx = new Transaction(doc, "Create Roof"))
{
    tx.Start();
    FootPrintRoof roof = doc.Create.NewFootPrintRoof(
        footprint, level, roofType, out ModelCurveArray curveMapping);

    // Set slope for each edge
    foreach (ModelCurve mc in curveMapping)
    {
        roof.set_DefinesSlope(mc, true);
        roof.set_SlopeAngle(mc, 0.5236); // 30 degrees in radians
    }
    tx.Commit();
}

Levels & Grids

C#
// Get all levels sorted by elevation
var levels = new FilteredElementCollector(doc)
    .OfClass(typeof(Level))
    .Cast<Level>()
    .OrderBy(l => l.Elevation)
    .ToList();

// Level properties
Level lvl = levels[0];
string name      = lvl.Name;
double elevation = lvl.Elevation;   // feet
double elev_mm   = elevation * 304.8;

// Create a new level
using (var tx = new Transaction(doc, "New Level"))
{
    tx.Start();
    Level newLevel = Level.Create(doc, 10.0);   // 10 feet elevation
    newLevel.Name = "Level 4";
    tx.Commit();
}

// Grids
var grids = new FilteredElementCollector(doc)
    .OfClass(typeof(Grid)).Cast<Grid>().ToList();

// Create a grid
using (var tx = new Transaction(doc, "New Grid"))
{
    tx.Start();
    Line gridLine = Line.CreateBound(new XYZ(0,0,0), new XYZ(0,20,0));
    Grid grid = Grid.Create(doc, gridLine);
    grid.Name = "A";
    tx.Commit();
}

Openings & Rooms

C#
// Create a rectangular opening in a wall
using (var tx = new Transaction(doc, "Create Opening"))
{
    tx.Start();
    Opening op = doc.Create.NewOpening(
        wall,
        new XYZ(2, 0, 0.5),    // lower-left corner
        new XYZ(4, 0, 2.5));   // upper-right corner
    tx.Commit();
}

// Rooms — get all rooms
var rooms = new FilteredElementCollector(doc)
    .OfCategory(BuiltInCategory.OST_Rooms)
    .Cast<Room>()
    .ToList();

foreach (var room in rooms)
{
    string name   = room.Name;
    string number = room.Number;
    double area   = room.Area;           // sq ft
    double areaSM = area * 0.0929;      // to sq metres

    // Room boundaries
    SpatialElementBoundaryOptions opts =
        new SpatialElementBoundaryOptions();
    IList<IList<BoundarySegment>> boundaries =
        room.GetBoundarySegments(opts);
}

Family Instances

C#
// Get a FamilySymbol (door type)
FamilySymbol sym = new FilteredElementCollector(doc)
    .OfClass(typeof(FamilySymbol))
    .OfCategory(BuiltInCategory.OST_Doors)
    .Cast<FamilySymbol>()
    .First();

using (var tx = new Transaction(doc, "Place Instance"))
{
    tx.Start();

    // Activate symbol if not already active
    if (!sym.IsActive) sym.Activate();

    // Place on a host (wall)
    FamilyInstance fi = doc.Create.NewFamilyInstance(
        new XYZ(5,0,0),      // location
        sym,               // symbol
        wall,              // host
        StructuralType.NonStructural);

    tx.Commit();
}

// Read family instance properties
FamilyInstance door = ...;
Family       family = door.Symbol.Family;
string         fname  = family.Name;
FamilySymbol   sym2   = door.Symbol;
string         sname  = sym2.Name;
Element        host   = door.Host;          // the wall
LocationPoint  lp     = door.Location as LocationPoint;
XYZ            pos    = lp.Point;
double         rot    = lp.Rotation;        // radians

Geometry Extraction

Access solids, faces and edges from any element using get_Geometry(). Always set ComputeReferences = true when dimensioning.

C#
var opts = new Options()
{
    View                    = doc.ActiveView,
    ComputeReferences       = true,    // REQUIRED for dimensioning!
    IncludeNonVisibleObjects = false
};

GeometryElement geomElem = element.get_Geometry(opts);

foreach (GeometryObject obj in geomElem)
{
    if (obj is Solid solid && solid.Volume > 0)
    {
        foreach (Face face in solid.Faces)
        {
            Reference faceRef  = face.Reference;     // for dimensions
            XYZ       normal   = face.ComputeNormal(new UV(.5,.5));
            double    area     = face.Area;           // sq ft
        }
        foreach (Edge edge in solid.Edges)
        {
            Curve edgeCurve = edge.AsCurve();
            double length   = edgeCurve.Length;
        }
    }
    else if (obj is GeometryInstance gi)
    {
        // FamilyInstance contains a GeometryInstance
        GeometryElement instGeom = gi.GetInstanceGeometry();
    }
}
