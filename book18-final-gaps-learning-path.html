ClassRebar — Layout Rules
Inherits: DB.Structure.Rebar
Namespace: Autodesk.Revit.DB.Structure
The professional way to place a full set of bars — create ONE Rebar element then apply a layout rule. Revit distributes the bars automatically. This is how Revit's own reinforcement tool works internally and produces proper bar sets that schedule correctly.
// Book 17 of 19 vs Book 18Book 17 loops to place individual bars (great for learning). Production add-ins use Layout Rules: place one Rebar, set the rule, Revit creates the full set with correct scheduling.
🟢 All 4 Layout Modes
// All 4 Rebar Layout Rules with real examples
using Autodesk.Revit.DB.Structure;

double mmToFt(double mm) =>
    UnitUtils.ConvertToInternalUnits(mm, UnitTypeId.Millimeters);

RebarBarType bt = new FilteredElementCollector(doc)
    .OfClass(typeof(RebarBarType)).Cast<RebarBarType>()
    .OrderBy(b => Math.Abs(b.BarNominalDiameter * 304.8 - 20)).First();

// ── Place ONE bar (position 0) ──
LocationCurve lc     = beam.Location as LocationCurve;
XYZ           bStart = lc.Curve.GetEndPoint(0);
XYZ           bEnd   = lc.Curve.GetEndPoint(1);
XYZ           dir    = (bEnd - bStart).Normalize();
XYZ           perp   = dir.CrossProduct(XYZ.BasisZ).Normalize();

double beamW  = mmToFt(300);
double beamH  = mmToFt(600);
double cvFt   = mmToFt(35);

XYZ off   = perp * (-beamW/2 + cvFt + mmToFt(10))
           + XYZ.BasisZ * (-beamH/2 + cvFt + mmToFt(10));
XYZ barS  = bStart + off + dir * cvFt;
XYZ barE  = bEnd   + off - dir * cvFt;

using (var tx = new Transaction(doc, "Rebar Layout"))
{
    tx.Start();

    Rebar rebar = Rebar.CreateFromCurves(
        doc, RebarStyle.Standard, bt,
        null, null, beam, XYZ.BasisZ,
        new List<Curve> { Line.CreateBound(barS, barE) },
        RebarHookOrientation.Left,
        RebarHookOrientation.Left);

    // ════ LAYOUT 1: Fixed Number ════
    // Place exactly 5 bars evenly distributed across beam width
    rebar.SetLayoutAsFixedNumber(
        5,                         // number of bars
        beamW - 2 * cvFt,          // array length (total span)
        true,                      // include first bar
        true,                      // include last bar
        true);                     // on normal side

    System.Diagnostics.Debug.WriteLine(
        $"Fixed: {rebar.NumberOfBarPositions} bars at {rebar.MaxSpacing*304.8:F0}mm");

    // ════ LAYOUT 2: Maximum Spacing ════
    // Fill the zone with bars — Revit calculates how many fit at max 150mm
    rebar.SetLayoutAsMaximumSpacing(
        mmToFt(150),               // maximum spacing
        beamW - 2 * cvFt,          // array length
        true, true, true);

    System.Diagnostics.Debug.WriteLine(
        $"MaxSpacing: {rebar.NumberOfBarPositions} bars");

    // ════ LAYOUT 3: Minimum Clear Spacing ════
    // Bars with 25mm minimum clear gap between them
    rebar.SetLayoutAsMinimumClearSpacing(
        mmToFt(25),                // minimum clear gap
        beamW - 2 * cvFt,
        true, true, true);

    // ════ LAYOUT 4: Single Bar (reset to just 1) ════
    rebar.SetLayoutAsSingleBar();

    tx.Commit();
}

// ── Read layout properties of any rebar ──
var bars = new FilteredElementCollector(doc)
    .OfCategory(BuiltInCategory.OST_Rebar)
    .Cast<Rebar>().ToList();

foreach (var bar in bars)
{
    RebarLayoutRule rule  = bar.LayoutRule;
    int             count = bar.NumberOfBarPositions;
    double          sp    = bar.MaxSpacing * 304.8;

    // Get transform for each bar position in the set
    for (int i = 0; i < count; i++)
    {
        Transform t = bar.GetShapeDrivenAccessor()
                          .GetBarPositionTransform(i);
        System.Diagnostics.Debug.WriteLine(
            $"Bar {i}: Rule={rule} Count={count} Sp={sp:F0}mm Pos={t.Origin}");
    }
}

MethodsGetBendData · RebarShapeDrivenAccessor · ComputeDrivingCurves
Namespace: Autodesk.Revit.DB.Structure
Access bend leg lengths, bend angles, and mandrel sizes for bar bending schedule generation. GetShapeDrivenAccessor() returns a RebarShapeDrivenAccessor that exposes ComputeDrivingCurves() for individual leg lengths.
🟢 Bend data extraction for bar bending schedules
// Extract bend data → CSV for bar bending schedule
public void ExtractBendData(Document doc)
{
    var bars = new FilteredElementCollector(doc)
        .OfCategory(BuiltInCategory.OST_Rebar)
        .Cast<Rebar>().ToList();

    var sb = new System.Text.StringBuilder();
    sb.AppendLine("Mark,Shape,Dia(mm),TotalLen(mm),Legs,A(mm),B(mm),C(mm),BendDia(mm)");

    foreach (var bar in bars)
    {
        string mark  = bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
                           ?.AsString() ?? "";
        RebarShape shape = doc.GetElement(bar.RebarShapeId) as RebarShape;
        string shpName   = shape?.Name ?? "Straight";
        double totalMm   = bar.get_Parameter(
            BuiltInParameter.REBAR_ELEM_LENGTH)?.AsDouble() * 304.8 ?? 0;

        RebarBarType bt  = doc.GetElement(bar.GetTypeId()) as RebarBarType;
        double dMm       = bt?.BarNominalDiameter * 304.8 ?? 0;
        double bendDiaMm = bt?.StandardBendDiameter * 304.8 ?? dMm * 4;

        // Get individual leg lengths from shape accessor
        RebarShapeDrivenAccessor acc = bar.GetShapeDrivenAccessor();
        IList<Curve> legs = acc?.ComputeDrivingCurves();
        var legsMm = legs?.Select(c => c.Length * 304.8).ToList()
                    ?? new List<double>();

        double A = legsMm.Count > 0 ? legsMm[0] : 0;
        double B = legsMm.Count > 1 ? legsMm[1] : 0;
        double C = legsMm.Count > 2 ? legsMm[2] : 0;

        sb.AppendLine($"{mark},{shpName},{dMm:F0},{totalMm:F0},"
                    + $"{legsMm.Count},{A:F0},{B:F0},{C:F0},{bendDiaMm:F0}");
    }

    string path = System.IO.Path.Combine(
        System.Environment.GetFolderPath(
            System.Environment.SpecialFolder.Desktop), "BendData.csv");
    System.IO.File.WriteAllText(path, sb.ToString());
    TaskDialog.Show("Bend Data", $"Exported {bars.Count} bars → {path}");
}

// Place A393 BRC mesh in a concrete slab
public void PlaceFabricSheet(Document doc, Floor slab)
{
    FabricSheetType fst = new FilteredElementCollector(doc)
        .OfClass(typeof(FabricSheetType)).Cast<FabricSheetType>()
        .FirstOrDefault(t => t.Name.Contains("A393")
                          || t.Name.Contains("A252"));

    if (fst == null) {
        TaskDialog.Show("Info",
            "Load a Fabric Reinforcement type from the Structural library first.");
        return;
    }

    BoundingBoxXYZ bb = slab.get_BoundingBox(null);
    if (bb == null) return;

    double cvFt    = UnitUtils.ConvertToInternalUnits(25, UnitTypeId.Millimeters);
    XYZ    origin  = new XYZ(bb.Min.X + cvFt, bb.Min.Y + cvFt, bb.Min.Z + cvFt);

    using (var tx = new Transaction(doc, "Fabric Sheet"))
    {
        tx.Start();
        FabricSheet sheet = FabricSheet.Create(
            doc, fst.Id, slab, origin, XYZ.BasisX);
        sheet.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)?.Set("FS-A393-BOT");
        tx.Commit();
    }

    // Read FabricSheetType properties
    System.Diagnostics.Debug.WriteLine($"Sheet: {fst.Name}");
    System.Diagnostics.Debug.WriteLine(
        $"  Major: Ø{fst.MajorWireDiameter*304.8:F1}mm @{fst.MajorWireSpacing*304.8:F0}mm");

        $"  Minor: Ø{fst.MinorWireDiameter*304.8:F1}mm @{fst.MinorWireSpacing*304.8:F0}mm");

ClassCeiling.Create
Inherits: HostObject → Element
Namespace: Autodesk.Revit.DB
Creates a ceiling from CurveLoops — identical API pattern to Floor.Create(). Use room boundaries as the profile. The Height Offset from Level parameter controls how far below the level the ceiling sits.
🟢 Create ceiling + batch create all room ceilings
// Ceiling.Create from room boundary + batch all rooms
public void CreateCeilingFromRoom(Document doc, Room room)
{
    CeilingType ct = new FilteredElementCollector(doc)
        .OfClass(typeof(CeilingType)).Cast<CeilingType>().First();

    var bOpts = new SpatialElementBoundaryOptions();
    var segs  = room.GetBoundarySegments(bOpts);
    if (segs.Count == 0) return;

    var loop = new CurveLoop();
    foreach (var seg in segs[0]) loop.Append(seg.GetCurve());

    using (var tx = new Transaction(doc, "Create Ceiling"))
    {
        tx.Start();

        // Identical to Floor.Create — just different element type
        Ceiling ceiling = Ceiling.Create(
            doc,
            new List<CurveLoop> { loop },
            ct.Id,
            room.LevelId);

        // Height offset below the level (negative = below)
        double off = UnitUtils.ConvertToInternalUnits(200, UnitTypeId.Millimeters);
        ceiling.get_Parameter(BuiltInParameter.CEILING_HEIGHTABOVELEVEL_PARAM)
               ?.Set(-off);

        tx.Commit();
    }
}

// ── Batch: create ceilings for ALL rooms on a level ──
public void BatchCeilings(Document doc, Level level)
{
    var rooms = new FilteredElementCollector(doc)
        .OfCategory(BuiltInCategory.OST_Rooms)
        .Cast<Room>()
        .Where(r => r.Area > 0 && r.LevelId == level.Id).ToList();

    CeilingType ct = new FilteredElementCollector(doc)
        .OfClass(typeof(CeilingType)).Cast<CeilingType>().First();
    int n = 0;

    using (var tx = new Transaction(doc, "Batch Ceilings"))
    {
        tx.Start();
        foreach (var room in rooms)
        {
            try {
                var bOpts = new SpatialElementBoundaryOptions();
                var segs  = room.GetBoundarySegments(bOpts);
                if (segs.Count == 0) continue;
                var loop = new CurveLoop();
                foreach (var seg in segs[0]) loop.Append(seg.GetCurve());
                Ceiling.Create(doc,
                    new List<CurveLoop>{ loop }, ct.Id, room.LevelId);
                n++;
            } catch { }
        }
        tx.Commit();
    }
    TaskDialog.Show("Done", $"Created {n} ceilings on {level.Name}");
}

        .OfClass(typeof(Group)).Cast<Group>()
        .Where(g => g.GroupType.Id == gt.Id).ToList();

        foreach (var g in instances)
        {
            ICollection<ElementId> members = g.UngroupMembers();
            System.Diagnostics.Debug.WriteLine(
                $"Ungrouped {members.Count} elements");
        }
        tx.Commit();
    }
}
ClassRevision · RevisionCloud
Namespace: Autodesk.Revit.DB
Represents a project revision. Revisions appear in the revision schedule on title blocks. RevisionClouds mark areas of change in views. Full workflow: Create Revision → add to Sheet → place RevisionCloud → Issue.
🟢 Complete revision workflow — create, cloud, issue, read
// Revision workflow: create → add to sheet → cloud → issue
public void FullRevisionWorkflow(Document doc,
    ViewSheet sheet, View view)
{
    Revision rev;

    using (var tx = new Transaction(doc, "Create Revision"))
    {
        tx.Start();

        // ── STEP 1: Create Revision ──
        rev              = Revision.Create(doc);
        rev.Description  = "Structural amendments per engineer review";
        rev.IssuedBy     = "Studio Architecture Ltd";
        rev.IssuedTo     = "Client: Acme Corp";
        rev.RevisionDate = "2024-06-01";
        rev.NumberType   = RevisionNumberType.Numeric;  // 1,2,3 or Alphanumeric A,B,C

        // ── STEP 2: Add to sheet (appears in title block table) ──
        var ids = sheet.GetAdditionalRevisionIds().ToList();
        ids.Add(rev.Id);
        sheet.SetAdditionalRevisionIds(ids);

        tx.Commit();
    }

    using (var tx = new Transaction(doc, "Revision Cloud"))
    {
        tx.Start();

        // ── STEP 3: Place RevisionCloud in a view ──
        var loop = new CurveLoop();
        loop.Append(Line.CreateBound(new XYZ(0,0,0),new XYZ(5,0,0)));
        loop.Append(Line.CreateBound(new XYZ(5,0,0),new XYZ(5,3,0)));
        loop.Append(Line.CreateBound(new XYZ(5,3,0),new XYZ(0,3,0)));
        loop.Append(Line.CreateBound(new XYZ(0,3,0),new XYZ(0,0,0)));

        RevisionCloud cloud = RevisionCloud.Create(
            doc, view, rev.Id,
            new List<CurveLoop> { loop });
        cloud.get_Parameter(BuiltInParameter.ALL_MODEL_COMMENTS)
             ?.Set("Wall moved 300mm N per RFI-042");

        tx.Commit();
    }

    using (var tx = new Transaction(doc, "Issue Revision"))
    {
        tx.Start();
        rev.Issued = true;   // locks it — no more editing
        tx.Commit();
    }

    // ── Read all revisions on a sheet ──
    foreach (var rid in sheet.GetAllRevisionIds())
    {
        Revision r = doc.GetElement(rid) as Revision;
        System.Diagnostics.Debug.WriteLine(
            $"{r?.RevisionDate} | {r?.Description} | Issued={r?.Issued}");
    }
}
ClassSpatialFieldManager · AnalysisResultSchema
Namespace: Autodesk.Revit.DB.Analysis
Creates custom analysis display overlays on element faces in Revit 3D views — colour-mapped results like heat flux, stress contours, or daylighting. Used when a plugin needs to display numerical analysis results visually on the model geometry.
🟢 Display custom analysis results on wall faces
// Colour-map custom heat flux values on exterior wall faces
using Autodesk.Revit.DB.Analysis;

public void ShowAnalysisOverlay(Document doc, View3D view3D)
{
    // ── STEP 1: Get or create SpatialFieldManager for the view ──
    SpatialFieldManager sfm =
        SpatialFieldManager.GetSpatialFieldManager(view3D)
        ?? SpatialFieldManager.CreateSpatialFieldManager(view3D, 1);

    // ── STEP 2: Register result schema ──
    var schema = new AnalysisResultSchema("HeatFlux", "W/m²");
    schema.SetUnits(new List<string> { "W/m²" },
                    new List<double> { 1.0 });
    int schIdx = sfm.RegisterResult(schema);

    // ── STEP 3: For each exterior wall face — upload values ──
    var walls = new FilteredElementCollector(doc)
        .OfClass(typeof(Wall)).Cast<Wall>()
        .Where(w => w.WallType.Function == WallFunction.Exterior).ToList();

    var opts = new Options { ComputeReferences = true };
    foreach (var wall in walls)
    {
        foreach (GeometryObject obj in wall.get_Geometry(opts))
        {
            if (!(obj is Solid solid)) continue;
            foreach (Face face in solid.Faces)
            {
                XYZ n = face.ComputeNormal(new UV(.5,.5));
                if (Math.Abs(n.Z) > 0.3 || face.Reference == null) continue;

                int primIdx = sfm.AddSpatialFieldPrimitive(face.Reference);
                var uvPts = new FieldDomainPointsByUV();
                var vals  = new FieldValues();
                BoundingBoxUV bb = face.GetBoundingBox();

                // 5×5 grid sample
                for (int iu = 0; iu < 5; iu++)
                for (int iv = 0; iv < 5; iv++)
                {
                    double u = bb.Min.U + (bb.Max.U-bb.Min.U)*iu/4;
                    double v = bb.Min.V + (bb.Max.V-bb.Min.V)*iv/4;
                    uvPts.Add(new UV(u,v));
                    vals.Add(new ValueAtPoint(
                        new List<double> { 5.0 + v * 15.0 })); // fake heat flux
                }
                sfm.UpdateSpatialFieldPrimitive(primIdx, uvPts, vals, schIdx);
            }
        }
    }

    // ── STEP 4: Create + apply AnalysisDisplayStyle ──
    using (var tx = new Transaction(doc, "Analysis Style"))
    {
        tx.Start();
        AnalysisDisplayStyle style =
            AnalysisDisplayStyle.CreateAnalysisDisplayStyle(
                doc, "Heat Flux Style",
                new AnalysisDisplayColoredSurfaceSettings(),
                new AnalysisDisplayColorSettings(),
                new AnalysisDisplayLegendSettings());
        view3D.AnalysisDisplayStyleId = style.Id;
        tx.Commit();
    }
}

        string status = lt?.GetLinkedFileStatus().ToString() ?? "—";
        System.Diagnostics.Debug.WriteLine(
            $"CAD: {name} | Linked={isLink} | Status={status}");
    }
}

// ── EXPLODE import into Revit lines ──
public void ExplodeCAD(Document doc, ImportInstance cad)
{
    using (var tx = new Transaction(doc, "Explode CAD"))
    {
        tx.Start();
        ICollection<ElementId> ids =
            ImportInstance.ExplodeAndDelete(doc, cad.Id);
        System.Diagnostics.Debug.WriteLine(
            $"Exploded → {ids.Count} elements");
        tx.Commit();
    }
}
