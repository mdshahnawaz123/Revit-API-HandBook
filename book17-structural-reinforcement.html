// DB.Structure — Fundamentals

Rebar Fundamentals

Core classes, enums, and helper patterns used across all reinforcement workflows.

ClassRebar — Core placement class
Inherits: Element
Namespace: Autodesk.Revit.DB.Structure
The primary class for placing individual reinforcement bars. Every reinforcement bar — whether bottom, top, stirrup, or tie — is a Rebar element. Bars are created from curves, a bar type, optional hook types, a host structural element, and a normal vector defining the bar's plane.
🟢 Essential Helper — Get Bar Type by Diameter
// Helpers used throughout ALL examples in this book
using Autodesk.Revit.DB.Structure;

// ── Get RebarBarType by nominal diameter (mm) ──
RebarBarType GetBarType(Document doc, double diamMm)
{
    return new FilteredElementCollector(doc)
        .OfClass(typeof(RebarBarType))
        .Cast<RebarBarType>()
        .OrderBy(b => Math.Abs(b.BarNominalDiameter * 304.8 - diamMm))
        .FirstOrDefault();
}

// ── Get hook type (90°, 135°, 180°) ──
RebarHookType GetHookType(Document doc, double angleDeg)
{
    return new FilteredElementCollector(doc)
        .OfClass(typeof(RebarHookType))
        .Cast<RebarHookType>()
        .OrderBy(h => Math.Abs(h.HookAngle * (180 / Math.PI) - angleDeg))
        .FirstOrDefault();
}

// ── Get RebarShape by name (straight, stirrup, U-bar etc.) ──
RebarShape GetShape(Document doc, string nameContains = "Straight")
{
    return new FilteredElementCollector(doc)
        .OfClass(typeof(RebarShape))
        .Cast<RebarShape>()
        .FirstOrDefault(s => s.Name.Contains(nameContains));
}

// ── Convert mm to feet ──
double mmToFt(double mm) =>
    UnitUtils.ConvertToInternalUnits(mm, UnitTypeId.Millimeters);

// ── Single bar placement wrapper ──
Rebar PlaceBar(Document doc, Element host,
    RebarBarType barType, Curve curve,
    XYZ normal, RebarStyle style = RebarStyle.Standard,
    RebarHookType startHook = null, RebarHookType endHook = null)
{
    return Rebar.CreateFromCurves(
        doc, style, barType,
        startHook, endHook,
        host, normal,
        new List<Curve> { curve },
        RebarHookOrientation.Left,
        RebarHookOrientation.Left);
}

// ── RebarStyle values ──
// RebarStyle.Standard       — normal straight or shaped bar
// RebarStyle.StirrupTie     — closed stirrup/tie (links)
// RebarStyle.Unknown        — not used in new code

// Beam Reinforcement

Beam — Bottom · Top · Stirrups

Complete beam reinforcement: bottom longitudinal bars, top longitudinal bars, and vertical stirrups (links) — with correct cover distances and hook configuration.

B

Beam reinforcement conventions

Bottom bars resist sagging moment (tension at bottom). Top bars resist hogging (tension at top — over supports). Stirrups resist shear and hold the cage together. All bars placed parallel to the beam centreline except stirrups. Cover = 35mm for internal beams (Eurocode), 40mm external. Normal vector for longitudinal bars = perpendicular to beam axis, in the XZ plane.

Bottom Longitudinal Bars
🟢 Place bottom tension bars in a concrete beam
// Bottom bars — 3T20 at beam soffit with 35mm cover
/// <summary>
/// Place bottom longitudinal reinforcement bars in a concrete beam.
/// 3 x Ø20mm bottom bars at 35mm cover from soffit and sides.
/// </summary>
public void ReinforcBeamBottom(Document doc, FamilyInstance beam)
{
    // ── Get beam geometry ──
    LocationCurve lc       = beam.Location as LocationCurve;
    Line          beamLine = lc?.Curve as Line;
    if (beamLine == null) return;

    XYZ start = beamLine.GetEndPoint(0);
    XYZ end   = beamLine.GetEndPoint(1);
    XYZ dir   = (end - start).Normalize();

    // ── Read beam cross-section dimensions ──
    double beamWidthFt  = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble()
        ?? mmToFt(300);
    double beamDepthFt  = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_HW)?.AsDouble()
        ?? mmToFt(600);

    // ── Bar configuration ──
    double coverFt   = mmToFt(35);  // Eurocode nominal cover
    double barDiamFt = mmToFt(20);  // Ø20 bars
    int    numBars   = 3;             // 3T20 bottom

    RebarBarType bt20 = GetBarType(doc, 20);
    RebarHookType hook90 = GetHookType(doc, 90);

    // ── Calculate bar positions across bottom of beam ──
    // centreZ = beam bottom + cover + half bar diameter
    double barZOffset = -beamDepthFt / 2 + coverFt + barDiamFt / 2;

    // Distribute bars across width
    double availWidth = beamWidthFt - 2 * (coverFt + barDiamFt / 2);
    double spacing    = numBars > 1 ? availWidth / (numBars - 1) : 0;

    // Normal vector for longitudinal bar plane = vertical (Z)
    // — perpendicular to beam axis, in vertical plane
    XYZ normal = XYZ.BasisZ;

    using (var tx = new Transaction(doc, "Beam Bottom Bars"))
    {
        tx.Start();

        for (int i = 0; i < numBars; i++)
        {
            double xOffset = -availWidth / 2 + i * spacing;

            // Build offset transform: shift bar to correct position
            XYZ perp   = dir.CrossProduct(XYZ.BasisZ).Normalize();
            XYZ barOff = perp * xOffset + XYZ.BasisZ * barZOffset;

            XYZ barStart = start + barOff;
            XYZ barEnd   = end   + barOff;

            // Shorten bar by cover at each end (bar stops before the face)
            XYZ s = barStart + dir * coverFt;
            XYZ e = barEnd   - dir * coverFt;

            Line barCurve = Line.CreateBound(s, e);

            Rebar bar = Rebar.CreateFromCurves(
                doc,
                RebarStyle.Standard,
                bt20,
                hook90,   // start hook 90°
                hook90,   // end hook 90°
                beam,
                normal,
                new List<Curve> { barCurve },
                RebarHookOrientation.Left,
                RebarHookOrientation.Left);

            // Set bar mark
            bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"B{i+1}-BOT-Ø20");
        }

        tx.Commit();
    }
}
Top Longitudinal Bars
🟢 Place top compression/hogging bars
// Top bars — 2T16 at top of beam with 35mm cover
public void ReinforcBeamTop(Document doc, FamilyInstance beam)
{
    LocationCurve lc    = beam.Location as LocationCurve;
    Line          line  = lc?.Curve as Line;
    if (line == null) return;

    XYZ start = line.GetEndPoint(0);
    XYZ end   = line.GetEndPoint(1);
    XYZ dir   = (end - start).Normalize();

    double beamWidthFt = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble()
        ?? mmToFt(300);
    double beamDepthFt = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_HW)?.AsDouble()
        ?? mmToFt(600);

    double coverFt   = mmToFt(35);
    double barDiamFt = mmToFt(16);
    int    numBars   = 2;   // 2T16 top bars

    RebarBarType bt16 = GetBarType(doc, 16);

    // ++ for top: offset UPWARD from beam centre
    double barZOffset = +beamDepthFt / 2 - coverFt - barDiamFt / 2;
    double availWidth = beamWidthFt - 2 * (coverFt + barDiamFt / 2);
    double spacing    = numBars > 1 ? availWidth / (numBars - 1) : 0;
    XYZ    perp      = dir.CrossProduct(XYZ.BasisZ).Normalize();

    using (var tx = new Transaction(doc, "Beam Top Bars"))
    {
        tx.Start();
        for (int i = 0; i < numBars; i++)
        {
            double xOff  = -availWidth / 2 + i * spacing;
            XYZ    off   = perp * xOff + XYZ.BasisZ * barZOffset;
            XYZ    s     = start + off + dir * coverFt;
            XYZ    e     = end   + off - dir * coverFt;
            Rebar  bar   = PlaceBar(doc, beam, bt16,
                Line.CreateBound(s, e), XYZ.BasisZ);
            bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"B{i+1}-TOP-Ø16");
        }
        tx.Commit();
    }
}
Stirrups / Shear Links
🟢 Place vertical closed stirrups along beam
// Stirrups — Ø10 links at 150mm centres along full beam length
public void ReinforcBeamStirrups(Document doc, FamilyInstance beam,
    double spacingMm = 150, double linkDiamMm = 10)
{
    LocationCurve lc   = beam.Location as LocationCurve;
    Line          line = lc?.Curve as Line;
    if (line == null) return;

    XYZ start     = line.GetEndPoint(0);
    XYZ end       = line.GetEndPoint(1);
    XYZ dir       = (end - start).Normalize();
    double lenFt  = line.Length;

    double coverFt    = mmToFt(35);
    double spacingFt  = mmToFt(spacingMm);
    double dFt        = mmToFt(linkDiamMm);

    double beamW = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble() ?? mmToFt(300);
    double beamH = beam.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_HW)?.AsDouble() ?? mmToFt(600);

    RebarBarType  bt10 = GetBarType(doc, linkDiamMm);
    RebarHookType h135 = GetHookType(doc, 135);  // 135° hook for seismic

    // Inner dimensions of stirrup
    double sw = beamW - 2 * coverFt;  // stirrup inner width
    double sh = beamH - 2 * coverFt;  // stirrup inner height

    // Perpendicular to beam axis (horizontal, for stirrup plane)
    XYZ perp = dir.CrossProduct(XYZ.BasisZ).Normalize();

    int numLinks = (int)Math.Floor((lenFt - 2 * coverFt) / spacingFt) + 1;

    using (var tx = new Transaction(doc, "Beam Stirrups"))
    {
        tx.Start();

        for (int i = 0; i < numLinks; i++)
        {
            double dist = coverFt + i * spacingFt;
            if (dist > lenFt - coverFt) break;

            // Centre of stirrup along beam axis
            XYZ ctr = start + dir * dist;

            // 4 corners of the closed stirrup rectangle
            // p0=bottom-left, p1=bottom-right, p2=top-right, p3=top-left
            XYZ p0 = ctr - perp * sw / 2 - XYZ.BasisZ * sh / 2;
            XYZ p1 = ctr + perp * sw / 2 - XYZ.BasisZ * sh / 2;
            XYZ p2 = ctr + perp * sw / 2 + XYZ.BasisZ * sh / 2;
            XYZ p3 = ctr - perp * sw / 2 + XYZ.BasisZ * sh / 2;

            // Stirrup = 4 lines forming closed rectangle
            var curves = new List<Curve>
            {
                Line.CreateBound(p0, p1),  // bottom
                Line.CreateBound(p1, p2),  // right
                Line.CreateBound(p2, p3),  // top
                Line.CreateBound(p3, p0)   // left
            };

            // Normal = beam direction (stirrup plane perpendicular to beam)
            Rebar stirrup = Rebar.CreateFromCurves(
                doc,
                RebarStyle.StirrupTie,   // ← MUST be StirrupTie for closed links
                bt10,
                h135, h135,              // 135° hooks for seismic
                beam,
                dir,                     // ← normal = beam direction
                curves,
                RebarHookOrientation.Left,
                RebarHookOrientation.Right);

            stirrup.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
                   ?.Set($"L{i+1}-Ø{linkDiamMm}@{spacingMm}");
        }

        tx.Commit();
    }
}
Complete Beam Command
🟢 One command — full beam cage: bottom + top + stirrups
// IExternalCommand — reinforce all selected beams automatically
[Transaction(TransactionMode.Manual)]
public class ReinforcBeamsCommand : IExternalCommand
{
    public Result Execute(ExternalCommandData cd, ref string msg, ElementSet els)
    {
        Document doc   = cd.Application.ActiveUIDocument.Document;
        UIDocument uid  = cd.Application.ActiveUIDocument;

        // Let user pick beams
        IList<Reference> refs;
        try
        {
            refs = uid.Selection.PickObjects(
                Autodesk.Revit.UI.Selection.ObjectType.Element,
                new BeamSelectionFilter(),
                "Select concrete beams to reinforce");
        }
        catch (Autodesk.Revit.Exceptions.OperationCanceledException)
        {
            return Result.Cancelled;
        }

        int count = 0;
        foreach (var r in refs)
        {
            FamilyInstance beam = doc.GetElement(r) as FamilyInstance;
            if (beam == null) continue;

            ReinforcBeamBottom(doc, beam);       // 3T20 bottom
            ReinforcBeamTop(doc, beam);           // 2T16 top
            ReinforcBeamStirrups(doc, beam,
                spacingMm: 150, linkDiamMm: 10); // Ø10@150
            count++;
        }

        TaskDialog.Show("Done",
            $"Reinforced {count} beam(s) with:\n" +
            "  3T20 bottom bars\n  2T16 top bars\n  Ø10 stirrups @150mm");

        return Result.Succeeded;
    }
}

public class BeamSelectionFilter : ISelectionFilter
{
    public bool AllowElement(Element e)
        => e is FamilyInstance fi
           && fi.StructuralType == StructuralType.Beam;
    public bool AllowReference(Reference r, XYZ p) => true;
}

// Column Reinforcement

Column — Longitudinal Bars · Ties

Reinforcement cage for rectangular concrete columns: corner bars + intermediate bars + rectangular ties at specified spacing.

C

Column reinforcement conventions

Longitudinal bars run vertically the full height of the column. They are positioned around the perimeter with 40mm cover (external column). Ties are horizontal closed links at vertical spacing — typically 150–200mm. The column cage is assembled from 4 corner bars + intermediate side bars + ties. Normal vector for vertical bars = horizontal (XY plane). Normal for ties = vertical (Z axis).

Longitudinal Column Bars
🟢 8T20 longitudinal bars around column perimeter
// Column longitudinal — 8×Ø20 perimeter bars at 40mm cover
public void ReinforcColumnLongitudinal(Document doc,
    FamilyInstance column, int numBars = 8, double barDiamMm = 20)
{
    // ── Get column geometry ──
    LocationPoint lp  = column.Location as LocationPoint;
    if (lp == null) return;
    XYZ colBase = lp.Point;

    // Get column height from base to top level
    double colHeightFt = column.get_Parameter(
        BuiltInParameter.INSTANCE_LENGTH_PARAM)?.AsDouble()
        ?? mmToFt(3000);

    // Get cross-section dimensions
    double colBFt = column.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble()
        ?? mmToFt(400);
    double colHFt = column.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_HW)?.AsDouble()
        ?? mmToFt(400);

    double coverFt   = mmToFt(40);
    double dFt        = mmToFt(barDiamMm);
    double innerW     = colBFt - 2 * (coverFt + dFt / 2);
    double innerH     = colHFt - 2 * (coverFt + dFt / 2);

    RebarBarType bt = GetBarType(doc, barDiamMm);

    // ── Calculate bar positions around perimeter ──
    // For 8 bars: 4 corners + 1 intermediate per long side
    // Layout: 
    //   (0)-----(1)-----(2)   ← top
    //    |               |
    //   (7)             (3)   ← sides
    //    |               |
    //   (6)-----(5)-----(4)   ← bottom
    double x0 = colBase.X - innerW / 2;
    double x1 = colBase.X + innerW / 2;
    double y0 = colBase.Y - innerH / 2;
    double y1 = colBase.Y + innerH / 2;

    var positions = new List<XYZ>
    {
        new XYZ(x0,               y1, colBase.Z),  // TL corner
        new XYZ((x0+x1)/2,       y1, colBase.Z),  // T mid
        new XYZ(x1,               y1, colBase.Z),  // TR corner
        new XYZ(x1, (y0+y1)/2,       colBase.Z),  // R mid
        new XYZ(x1,               y0, colBase.Z),  // BR corner
        new XYZ((x0+x1)/2,       y0, colBase.Z),  // B mid
        new XYZ(x0,               y0, colBase.Z),  // BL corner
        new XYZ(x0, (y0+y1)/2,       colBase.Z),  // L mid
    };

    using (var tx = new Transaction(doc, "Column Longitudinal"))
    {
        tx.Start();

        for (int i = 0; i < Math.Min(numBars, positions.Count); i++)
        {
            XYZ  pos  = positions[i];
            XYZ  ptS  = new XYZ(pos.X, pos.Y, colBase.Z + coverFt);
            XYZ  ptE  = new XYZ(pos.X, pos.Y, colBase.Z + colHeightFt - coverFt);
            Line barL = Line.CreateBound(ptS, ptE);

            // Normal for vertical bar = horizontal (perpendicular to bar)
            // Use XYZ.BasisX for bars in YZ plane, BasisY for bars in XZ plane
            XYZ normal = Math.Abs(pos.X - colBase.X) >
                         Math.Abs(pos.Y - colBase.Y)
                ? XYZ.BasisY   // bar on left/right face: normal = Y
                : XYZ.BasisX;  // bar on front/back face: normal = X

            Rebar bar = Rebar.CreateFromCurves(
                doc,
                RebarStyle.Standard,
                bt,
                null, null,   // no hooks on vertical column bars
                column,
                normal,
                new List<Curve> { barL },
                RebarHookOrientation.Left,
                RebarHookOrientation.Left);

            bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"VB{i+1}-Ø{barDiamMm}");
        }

        tx.Commit();
    }
}
Column Ties / Rectangular Links
🟢 Rectangular ties at specified vertical spacing
// Column ties — Ø10 rectangular links at 200mm vertical centres
public void ReinforcColumnTies(Document doc, FamilyInstance column,
    double spacingMm = 200, double tieDiamMm = 10)
{
    LocationPoint lp = column.Location as LocationPoint;
    if (lp == null) return;
    XYZ base  = lp.Point;

    double colH = column.get_Parameter(
        BuiltInParameter.INSTANCE_LENGTH_PARAM)?.AsDouble() ?? mmToFt(3000);
    double colB = column.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble() ?? mmToFt(400);
    double colD = column.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_HW)?.AsDouble() ?? mmToFt(400);

    double coverFt  = mmToFt(40);
    double dFt      = mmToFt(tieDiamMm);
    double spFt     = mmToFt(spacingMm);

    // Inner rectangle of tie
    double tw = colB - 2 * coverFt;
    double td = colD - 2 * coverFt;

    RebarBarType  bt  = GetBarType(doc, tieDiamMm);
    RebarHookType h90 = GetHookType(doc, 90);

    int numTies = (int)Math.Floor((colH - 2 * coverFt) / spFt) + 1;

    using (var tx = new Transaction(doc, "Column Ties"))
    {
        tx.Start();

        for (int i = 0; i < numTies; i++)
        {
            double z = base.Z + coverFt + i * spFt;

            // 4 corners of horizontal tie at elevation z
            XYZ p0 = new XYZ(base.X - tw/2, base.Y - td/2, z);
            XYZ p1 = new XYZ(base.X + tw/2, base.Y - td/2, z);
            XYZ p2 = new XYZ(base.X + tw/2, base.Y + td/2, z);
            XYZ p3 = new XYZ(base.X - tw/2, base.Y + td/2, z);

            var curves = new List<Curve>
            {
                Line.CreateBound(p0, p1),
                Line.CreateBound(p1, p2),
                Line.CreateBound(p2, p3),
                Line.CreateBound(p3, p0)
            };

            // Normal = Z (tie lies in horizontal plane)
            Rebar tie = Rebar.CreateFromCurves(
                doc, RebarStyle.StirrupTie,
                bt, h90, h90,
                column, XYZ.BasisZ,
                curves,
                RebarHookOrientation.Left,
                RebarHookOrientation.Right);

            tie.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"T{i+1}-Ø{tieDiamMm}@{spacingMm}");
        }
        tx.Commit();
    }
}

// Slab Reinforcement

Slab — AreaReinforcement · PathReinforcement · Punching

Two-way slab reinforcement: bottom mesh (X+Y), top mesh over supports, and punching shear bars around columns.

S

Slab reinforcement approach

Two-way slabs use AreaReinforcement for uniform mesh (bottom and top). Local punching zones and strip reinforcement use PathReinforcement. Bottom bars are placed first (lowest layer), then top bars. Cover = 25mm internal / 35mm external. Bar spacing typically 150–200mm for T12 or T16.

Slab AreaReinforcement — Two-Way Mesh
🟢 Bottom and top mesh for a two-way concrete slab
// Two-way slab mesh — T12@150 bottom + T12@150 top
public void ReinforcSlabMesh(Document doc, Floor slab,
    double barDiamMm = 12, double spacingMm = 150)
{
    RebarBarType bt = GetBarType(doc, barDiamMm);
    if (bt == null) return;

    // Get slab boundary as CurveArray
    Options opts = new Options { ComputeReferences = false };
    BoundingBoxXYZ bb = slab.get_BoundingBox(null);
    if (bb == null) return;

    // Extract slab boundary curves from geometry
    CurveArray boundary = new CurveArray();
    foreach (GeometryObject obj in slab.get_Geometry(opts))
    {
        if (obj is Solid solid)
        {
            foreach (Face face in solid.Faces)
            {
                XYZ n = face.ComputeNormal(new UV(.5,.5));
                if (Math.Abs(n.Z) > 0.9 && n.Z > 0)   // top face
                {
                    foreach (var loop in face.GetEdgesAsCurveLoops())
                        foreach (var c in loop)
                            boundary.Append(c);
                    break;
                }
            }
        }
    }
    if (boundary.Size == 0) return;

    using (var tx = new Transaction(doc, "Slab Mesh Reinf"))
    {
        tx.Start();

        // ── BOTTOM MESH ──
        AreaReinforcement botReinf = AreaReinforcement.Create(
            doc,
            slab,
            boundary,
            XYZ.BasisX,      // major direction = X
            bt.Id,            // major bar type
            ElementId.InvalidElementId,
            bt.Id,            // minor bar type (Y direction)
            ElementId.InvalidElementId);

        // Set spacing for both directions
        botReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_SPACING_BOTTOM_DIR_1)
            ?.Set(mmToFt(spacingMm));
        botReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_SPACING_BOTTOM_DIR_2)
            ?.Set(mmToFt(spacingMm));

        // Bottom cover
        botReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_BOTTOM_COVER_1)
            ?.Set(mmToFt(25));
        botReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_BOTTOM_COVER_2)
            ?.Set(mmToFt(25 + barDiamMm));  // 2nd layer sits above 1st

        botReinf.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
            ?.Set($"BOT-T{barDiamMm}@{spacingMm}");

        // ── TOP MESH ── (same boundary, different cover)
        AreaReinforcement topReinf = AreaReinforcement.Create(
            doc, slab, boundary,
            XYZ.BasisX, bt.Id,
            ElementId.InvalidElementId,
            bt.Id,
            ElementId.InvalidElementId);

        topReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_SPACING_TOP_DIR_1)
            ?.Set(mmToFt(spacingMm));
        topReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_SPACING_TOP_DIR_2)
            ?.Set(mmToFt(spacingMm));
        topReinf.get_Parameter(BuiltInParameter.REBAR_AREA_REINF_TOP_COVER_1)
            ?.Set(mmToFt(25));
        topReinf.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
            ?.Set($"TOP-T{barDiamMm}@{spacingMm}");

        tx.Commit();
    }

Punching Shear Zone
🟢 Radial punching shear bars around a column head
// Punching zone — radial T12 bars on 1.5d perimeter around column
public void ReinforcPunchingZone(Document doc,
    Floor slab, FamilyInstance column,
    double slabDepthMm = 250, double barDiamMm = 12)
{
    // Column centre
    XYZ colCtr = ((column.Location as LocationPoint)?.Point)
        ?? XYZ.Zero;

    // Effective depth d = slab - cover - bar diameter
    double d       = slabDepthMm - 25 - barDiamMm;
    double perimFt = mmToFt(1.5 * d);  // critical perimeter at 1.5d

    // Get column half-dimensions
    double cHalf = (double)(column.get_Parameter(
        BuiltInParameter.STRUCTURAL_SECTION_ISHAPE_BF)?.AsDouble()
        ?? mmToFt(400)) / 2;

    double slabElev = (slab.get_BoundingBox(null)?.Max.Z ?? colCtr.Z) - mmToFt(50);

    RebarBarType bt = GetBarType(doc, barDiamMm);

    // Place 8 radial bars at 45° intervals around column
    int numRadial = 8;
    double innerR = cHalf + mmToFt(50);   // start 50mm from column face
    double outerR = perimFt + cHalf;       // out to 1.5d perimeter

    using (var tx = new Transaction(doc, "Punching Reinf"))
    {
        tx.Start();

        for (int i = 0; i < numRadial; i++)
        {
            double angle = i * (2 * Math.PI / numRadial);
            double cosA  = Math.Cos(angle);
            double sinA  = Math.Sin(angle);

            // Bar runs radially outward from column face to 1.5d perimeter
            XYZ s = new XYZ(
                colCtr.X + cosA * innerR,
                colCtr.Y + sinA * innerR,
                slabElev);
            XYZ e = new XYZ(
                colCtr.X + cosA * outerR,
                colCtr.Y + sinA * outerR,
                slabElev);

            // Normal = perpendicular to bar in horizontal plane
            XYZ radialDir = new XYZ(cosA, sinA, 0);
            XYZ normal    = radialDir.CrossProduct(XYZ.BasisZ).Normalize();

            Rebar bar = PlaceBar(doc, slab, bt,
                Line.CreateBound(s, e), normal);

            bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"PZ{i+1}-Ø{barDiamMm}");
        }
        tx.Commit();
    }
}

// Foundation Reinforcement

Foundation — Pad · Strip · Pile Cap

Bottom mesh for pad footings, bottom bars for strip footings, and pile cap cage — all with correct 75mm cover to external face.

F

Foundation reinforcement conventions

Cover to external face of foundation = 75mm (concrete cast against earth). Pad footing: two-way bottom mesh spanning to column face. Strip footing: longitudinal bottom bars + transverse distribution bars. Pile cap: bottom bars spanning between pile heads + perimeter tie bars. Starter bars for column lap are separate and placed after the footing cage.

Pad Footing Reinforcement
🟢 Two-way bottom mesh + starter bars for isolated pad footing
// Pad footing — T16@150 two-way mesh + 4 T20 starter bars at 75mm cover
public void ReinforcPadFooting(Document doc,
    FamilyInstance footing,  // structural foundation family instance
    double meshBarDiamMm = 16,
    double meshSpacingMm = 150,
    double starterBarDiamMm = 20,
    double starterLapMm = 600)
{
    LocationPoint lp = footing.Location as LocationPoint;
    if (lp == null) return;
    XYZ ctr = lp.Point;

    // Read footing dimensions
    double fW = footing.get_Parameter(
        BuiltInParameter.STRUCTURAL_FOUNDATION_WIDTH)?.AsDouble() ?? mmToFt(1500);
    double fD = footing.get_Parameter(
        BuiltInParameter.STRUCTURAL_FOUNDATION_LENGTH)?.AsDouble() ?? mmToFt(1500);
    double fH = footing.get_Parameter(
        BuiltInParameter.STRUCTURAL_FOUNDATION_THICKNESS)?.AsDouble() ?? mmToFt(500);

    double coverFt  = mmToFt(75);  // 75mm cover to earth
    double mDFt     = mmToFt(meshBarDiamMm);
    double spFt     = mmToFt(meshSpacingMm);

    double btmZ     = ctr.Z + coverFt + mDFt / 2;  // first layer Z
    double btmZ2    = btmZ + mDFt + mmToFt(5);      // second layer Z

    RebarBarType btMesh    = GetBarType(doc, meshBarDiamMm);
    RebarBarType btStarter = GetBarType(doc, starterBarDiamMm);

    using (var tx = new Transaction(doc, "Pad Footing Reinf"))
    {
        tx.Start();

        // ── Bottom mesh — X direction (first layer) ──
        int numX = (int)Math.Floor((fD - 2 * coverFt) / spFt) + 1;
        for (int i = 0; i < numX; i++)
        {
            double y = ctr.Y - fD/2 + coverFt + i * spFt;
            XYZ s = new XYZ(ctr.X - fW/2 + coverFt, y, btmZ);
            XYZ e = new XYZ(ctr.X + fW/2 - coverFt, y, btmZ);
            var bar = PlaceBar(doc, footing, btMesh,
                Line.CreateBound(s, e), XYZ.BasisZ);
            bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
               ?.Set($"FBX{i+1}-Ø{meshBarDiamMm}");
        }

        // ── Bottom mesh — Y direction (second layer) ──
        int numY = (int)Math.Floor((fW - 2 * coverFt) / spFt) + 1;
        for (int i = 0; i < numY; i++)
        {
            double x = ctr.X - fW/2 + coverFt + i * spFt;
            XYZ s = new XYZ(x, ctr.Y - fD/2 + coverFt, btmZ2);
            XYZ e = new XYZ(x, ctr.Y + fD/2 - coverFt, btmZ2);
            PlaceBar(doc, footing, btMesh,
                Line.CreateBound(s, e), XYZ.BasisY)
              .get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
              ?.Set($"FBY{i+1}-Ø{meshBarDiamMm}");
        }

        // ── Starter bars — 4 corner starters projecting above footing ──
        double colBFt  = mmToFt(400) / 2 - mmToFt(50);  // approx column face
        double startLp = mmToFt(starterLapMm);
        double topZ    = ctr.Z + fH + startLp;         // project above footing
        double sBaseZ  = ctr.Z + coverFt;              // starts at footing bottom

        double[] xs = { ctr.X - colBFt, ctr.X + colBFt,
                         ctr.X + colBFt, ctr.X - colBFt };
        double[] ys = { ctr.Y - colBFt, ctr.Y - colBFt,
                         ctr.Y + colBFt, ctr.Y + colBFt };

        for (int i = 0; i < 4; i++)
        {
            XYZ s = new XYZ(xs[i], ys[i], sBaseZ);
            XYZ e = new XYZ(xs[i], ys[i], topZ);
            PlaceBar(doc, footing, btStarter,
                Line.CreateBound(s, e),
                (i == 0 || i == 2) ? XYZ.BasisX : XYZ.BasisY)
              .get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
              ?.Set($"ST{i+1}-Ø{starterBarDiamMm}");
        }

        tx.Commit();
    }
}
Strip Footing Reinforcement
🟢 Longitudinal + transverse bars for strip/wall footing
// Strip footing — 3T16 bottom longitudinal + T10 transverse @300mm
public void ReinforcStripFooting(Document doc, Wall footingWall)
{
    LocationCurve lc   = footingWall.Location as LocationCurve;
    Line          line = lc?.Curve as Line;
    if (line == null) return;

    XYZ    s     = line.GetEndPoint(0);
    XYZ    e     = line.GetEndPoint(1);
    XYZ    dir   = (e - s).Normalize();
    double lenFt  = line.Length;

    double fWidthFt = footingWall.Width;
    double fDepthFt = (mmToFt(450)); // typical strip depth
    double cvFt     = mmToFt(75);

    RebarBarType bt16 = GetBarType(doc, 16);
    RebarBarType bt10 = GetBarType(doc, 10);
    XYZ          perp = dir.CrossProduct(XYZ.BasisZ).Normalize();

    using (var tx = new Transaction(doc, "Strip Footing Reinf"))
    {
        tx.Start();

        // ── 3 bottom longitudinal bars ──
        double[] xOffsets = {
            -fWidthFt/2 + cvFt + mmToFt(8),
            0,
            +fWidthFt/2 - cvFt - mmToFt(8)
        };
        double botZ = s.Z + cvFt + mmToFt(8);  // bottom of footing + cover + r

        foreach (var xOff in xOffsets)
        {
            XYZ off   = perp * xOff + XYZ.BasisZ * (botZ - s.Z);
            XYZ barS  = s + off + dir * cvFt;
            XYZ barE  = e + off - dir * cvFt;
            PlaceBar(doc, footingWall, bt16,
                Line.CreateBound(barS, barE), XYZ.BasisZ)
              .get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
              ?.Set("SF-LONG-Ø16");
        }

        // ── Transverse distribution bars @300mm ──
        double spT = mmToFt(300);
        int    nT  = (int)((lenFt - 2 * cvFt) / spT) + 1;

        for (int i = 0; i < nT; i++)
        {
            double dist  = cvFt + i * spT;
            if (dist > lenFt - cvFt) break;

            XYZ ctrPt = s + dir * dist + XYZ.BasisZ * (botZ - s.Z + mmToFt(16));
            XYZ tS    = ctrPt - perp * (fWidthFt/2 - cvFt);
            XYZ tE    = ctrPt + perp * (fWidthFt/2 - cvFt);
            PlaceBar(doc, footingWall, bt10,
                Line.CreateBound(tS, tE), dir)
              .get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
              ?.Set($"SF-T{i+1}-Ø10");
        }

        tx.Commit();
    }
}

public void AutoMarkAllRebar(Document doc)
{
    var bars = new FilteredElementCollector(doc)
        .OfCategory(BuiltInCategory.OST_Rebar)
        .Cast<Rebar>()
        .OrderBy(b => b.Id.IntegerValue)
        .ToList();

    using (var tx = new Transaction(doc, "Auto Mark Rebar"))
    {
        tx.Start();

        // Group by diameter and style, then assign sequential marks
        var groups = bars
            .GroupBy(b => (GetDiamMm(doc, b), b.RebarStyle))
            .OrderBy(g => g.Key.Item1);

        int barMark = 1;
        foreach (var grp in groups)
        {
            (double diam, RebarStyle style) = grp.Key;
            string prefix = style == RebarStyle.StirrupTie ? "L" : "B";
            int    idx    = 1;

            foreach (var bar in grp)
            {
                string mark = $"{prefix}{barMark:D2}-Ø{diam:F0}";
                bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
                   ?.Set(mark);
                idx++;
            }
            barMark++;
        }
        tx.Commit();
    }
}

double GetDiamMm(Document doc, Rebar bar)

    var bt = doc.GetElement(bar.GetTypeId()) as RebarBarType;

// STEP 2: Create rebar schedule view

public ViewSchedule CreateRebarSchedule(Document doc)
{
    ViewSchedule sched;

    using (var tx = new Transaction(doc, "Rebar Schedule"))
    {
        tx.Start();

        sched = ViewSchedule.CreateSchedule(doc,
            new ElementId(BuiltInCategory.OST_Rebar));
        sched.Name = "Bar Bending Schedule";

        ScheduleDefinition def    = sched.Definition;
        IList<SchedulableField> sf = def.GetSchedulableFields();

        // Add key fields for a bar bending schedule
        var wantedBIPs = new[]
        {
            BuiltInParameter.ALL_MODEL_MARK,              // Bar Mark
            BuiltInParameter.REBAR_ELEM_BAR_DIAMETER,     // Diameter
            BuiltInParameter.REBAR_ELEM_LENGTH,            // Total Length
            BuiltInParameter.REBAR_NUMBER_OF_BARSINSET,    // No. of Bars
            BuiltInParameter.REBAR_ELEM_BAR_STYLE,         // Style
            BuiltInParameter.HOST_ID_PARAM,               // Host Element
            BuiltInParameter.REBAR_ELEM_VOLUME,            // Volume (for weight)
        };

        foreach (var bip in wantedBIPs)
        {
            var field = sf.FirstOrDefault(
                f => f.ParameterId == new ElementId(bip));
            if (field != null) def.AddField(field);
        }

        // Sort by mark
        var markField = def.GetField(0);
        var sortGrp   = new ScheduleSortGroupField(markField.FieldId)
        {
            SortOrder = ScheduleSortOrder.Ascending
        };
        def.SetSortGroupFields(
            new List<ScheduleSortGroupField> { sortGrp });

        tx.Commit();
    }
    return sched;

public void ExportRebarToCSV(Document doc, string csvPath)
{
    var bars = new FilteredElementCollector(doc)
        .OfCategory(BuiltInCategory.OST_Rebar)
        .Cast<Rebar>()
        .ToList();

    var sb = new System.Text.StringBuilder();
    sb.AppendLine("Mark,Diameter(mm),Length(mm),Count,Style,Volume(m³),Weight(kg),Host");

    // Steel density: 7850 kg/m³
    const double STEEL_DENSITY = 7850;

    foreach (var bar in bars.OrderBy(b =>
        b.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)?.AsString()))
    {
        string mark   = bar.get_Parameter(BuiltInParameter.ALL_MODEL_MARK)
                            ?.AsString() ?? "";
        double diam   = GetDiamMm(doc, bar);
        double lenMm  = bar.get_Parameter(BuiltInParameter.REBAR_ELEM_LENGTH)
                            ?.AsDouble() * 304.8 ?? 0;
        int    count  = bar.get_Parameter(BuiltInParameter.REBAR_NUMBER_OF_BARSINSET)
                            ?.AsInteger() ?? 1;
        string style  = bar.RebarStyle.ToString();
        double volM3  = bar.get_Parameter(BuiltInParameter.REBAR_ELEM_VOLUME)
                            ?.AsDouble() * 0.0283168 ?? 0;
        double weightKg = volM3 * STEEL_DENSITY;

        // Get host name
        var    hostId   = bar.get_Parameter(BuiltInParameter.HOST_ID_PARAM)
                             ?.AsElementId();
        string hostName = hostId != null
            ? doc.GetElement(hostId)?.Name ?? ""
            : "";

        sb.AppendLine($"{mark},{diam:F0},{lenMm:F0},{count}," +
                       $"{style},{volM3:F6},{weightKg:F2},{hostName}");
    }

    System.IO.File.WriteAllText(csvPath, sb.ToString());

    // Summary
    double totalWeight = bars
        .Sum(b => (b.get_Parameter(BuiltInParameter.REBAR_ELEM_VOLUME)
                    ?.AsDouble() * 0.0283168 ?? 0) * STEEL_DENSITY);

    TaskDialog.Show("Rebar Export",
        $"Exported {bars.Count} bar records\n" +
        $"Total steel weight: {totalWeight:F0} kg\n" +
        $"CSV: {csvPath}");
}
