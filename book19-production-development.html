// WPF + MVVM

WPF + MVVM for Revit Plugins

The Model-View-ViewModel pattern — the standard architecture for all professional Revit plugin UIs. Separates Revit API logic from WPF UI completely.

📐

MVVM Architecture in a Revit Plugin

Model = Revit data (elements, parameters) accessed via API. ViewModel = C# class with properties and commands — completely independent of Revit API. View = XAML file — binds to ViewModel properties. The ViewModel never knows about WPF controls; the View never calls the Revit API directly. They communicate via ExternalEvent.

StudioTools/ ├── Commands/ │ └── OpenPanelCommand.cs ← IExternalCommand entry point ├── ViewModels/ │ ├── RoomInspectorViewModel.cs ← ViewModel (no Revit refs) │ └── ViewModelBase.cs ← INotifyPropertyChanged base ├── Views/ │ ├── RoomInspectorPane.xaml ← WPF UserControl (pure XAML) │ └── RoomInspectorPane.xaml.cs ← Code-behind (minimal) ├── Models/ │ └── RoomData.cs ← Plain C# data class (no Revit API) ├── Services/ │ ├── RevitDataService.cs ← Wraps Revit API calls │ └── ExternalEventBridge.cs ← IExternalEventHandler + ExternalEvent └── Application.cs ← IExternalApplication
InterfaceINotifyPropertyChanged — ViewModelBase
The foundation of WPF data binding. Any property change on the ViewModel automatically updates the bound WPF control. Implement once in a base class, inherit everywhere.
🟢 ViewModelBase + CallerMemberName pattern
// ViewModelBase.cs — implement once, inherit everywhere
using System.ComponentModel;
using System.Runtime.CompilerServices;

namespace StudioTools.ViewModels
{
    /// Base class for all ViewModels.
    /// Inherit this to get automatic WPF data binding.
    public abstract class ViewModelBase : INotifyPropertyChanged
    {
        public event PropertyChangedEventHandler PropertyChanged;

        // Call this inside any property setter to notify WPF
        protected void OnPropertyChanged(
            [CallerMemberName] string name = null)
            => PropertyChanged?.Invoke(this,
                new PropertyChangedEventArgs(name));

        // Set backing field and raise event in one call
        protected bool SetField<T>(ref T field, T value,
            [CallerMemberName] string name = null)
        {
            if (EqualityComparer<T>.Default.Equals(field, value)) return false;
            field = value;
            OnPropertyChanged(name);
            return true;
        }
    }
}

// ── Usage in a ViewModel ──
public class RoomInspectorViewModel : ViewModelBase
{
    private string _roomName;
    private double _roomArea;
    private bool   _isLoading;

    // WPF binds to these properties
    public string RoomName
    {
        get => _roomName;
        set => SetField(ref _roomName, value);  // auto raises PropertyChanged
    }

    public double RoomArea
    {
        get => _roomArea;
        set => SetField(ref _roomArea, value);
    }

    public bool IsLoading
    {
        get => _isLoading;
        set
        {
            SetField(ref _isLoading, value);
            OnPropertyChanged(nameof(LoadingVisibility));  // update derived
        }
    }

    // Computed property — no backing field, notified manually
    public System.Windows.Visibility LoadingVisibility
        => IsLoading ? System.Windows.Visibility.Visible
                     : System.Windows.Visibility.Collapsed;

    // Formatted display — recomputed when RoomArea changes
    public string RoomAreaDisplay => $"{RoomArea:F1} m²";
}
ClassRelayCommand — ICommand Implementation
ICommand implementation that binds WPF buttons to ViewModel methods. RelayCommand wraps an Action (execute) and optional Func<bool> (canExecute). When canExecute returns false, the button is automatically disabled.
🟢 RelayCommand + ViewModel button binding
// RelayCommand.cs + ViewModel binding + XAML
// ── RelayCommand.cs ──
using System.Windows.Input;

public class RelayCommand : ICommand
{
    private readonly Action<object>       _execute;
    private readonly Func<object, bool> _canExecute;

    public RelayCommand(Action<object> execute,
        Func<object, bool> canExecute = null)
    {
        _execute    = execute    ?? throw new ArgumentNullException(nameof(execute));
        _canExecute = canExecute;
    }

    public event EventHandler CanExecuteChanged
    {
        add    => System.Windows.Input.CommandManager.RequerySuggested += value;
        remove => System.Windows.Input.CommandManager.RequerySuggested -= value;
    }

    public bool   CanExecute(object p) => _canExecute?.Invoke(p) ?? true;
    public void   Execute(object p)    => _execute(p);

    // Force WPF to recheck CanExecute on all commands
    public static void RaiseCanExecuteChanged()
        => System.Windows.Input.CommandManager.InvalidateRequerySuggested();
}

// Parameterless shortcut
public class RelayCommand<T> : ICommand
{
    private readonly Action<T>       _execute;
    private readonly Func<T, bool> _canExecute;
    public event EventHandler CanExecuteChanged
    {
        add    => CommandManager.RequerySuggested += value;
        remove => CommandManager.RequerySuggested -= value;
    }
    public RelayCommand(Action<T> exec, Func<T,bool> canExec = null)
    { _execute = exec; _canExecute = canExec; }
    public bool CanExecute(object p)
        => _canExecute?.Invoke((T)p) ?? true;
    public void Execute(object p) => _execute((T)p);
}

// ── ViewModel with commands ──
public class RoomInspectorViewModel : ViewModelBase
{
    private ObservableCollection<RoomData> _rooms
        = new ObservableCollection<RoomData>();
    private RoomData _selectedRoom;
    private bool _hasRooms;

    public ObservableCollection<RoomData> Rooms
    {
        get => _rooms;
        set => SetField(ref _rooms, value);
    }

    public RoomData SelectedRoom
    {
        get => _selectedRoom;
        set
        {
            SetField(ref _selectedRoom, value);
            // Notify that ExportCommand.CanExecute may have changed
            RelayCommand.RaiseCanExecuteChanged();
        }
    }

    // Commands bound to buttons in XAML
    public ICommand RefreshCommand { get; }
    public ICommand ExportCommand  { get; }
    public ICommand SelectInRevit  { get; }

    public RoomInspectorViewModel()
    {
        RefreshCommand = new RelayCommand(_ => OnRefresh());
        ExportCommand  = new RelayCommand(
            _ => OnExport(),
            _ => SelectedRoom != null);  // disabled if nothing selected
        SelectInRevit  = new RelayCommand(
            _ => OnSelectInRevit(),
            _ => SelectedRoom != null);
    }

    private void OnRefresh()
    {
        // Raise external event to load rooms from Revit on the main thread
        StudioApplication.LoadRoomsEvent.Raise();
    }

    private void OnExport()
    {
        // Export selected room data — no Revit API needed
        System.Windows.MessageBox.Show(
            $"Exporting {SelectedRoom?.Name}");
    }

    private void OnSelectInRevit()
    {
        StudioApplication.SelectElementHandler.ElementId =
            SelectedRoom?.RevitId;
        StudioApplication.SelectElementEvent.Raise();
    }

    // Called by ExternalEventHandler after Revit loads data
    public void UpdateRooms(IEnumerable<RoomData> rooms)
    {
        Rooms.Clear();
        foreach (var r in rooms) Rooms.Add(r);
    }
}

PatternMVVM + Revit ExternalEvent Bridge
The complete pattern connecting a WPF ViewModel to Revit's API via ExternalEvents. The ViewModel never touches the Revit API directly — it only calls ExternalEvent.Raise(). The handler reads/writes Revit, then updates the ViewModel on the main WPF thread.
🟢 Full MVVM + Revit Bridge
// Complete MVVM bridge — ViewModel → ExternalEvent → Revit → ViewModel
// ── RoomData.cs (Model — pure C#, no Revit refs) ──
public class RoomData
{
    public ElementId  RevitId    { get; set; }
    public string     Name       { get; set; }
    public string     Number     { get; set; }
    public string     Department { get; set; }
    public double     AreaSqM    { get; set; }
    public string     Level      { get; set; }
    public string     AreaDisplay => $"{AreaSqM:F1} m²";
}

// ── LoadRoomsHandler.cs (IExternalEventHandler) ──
public class LoadRoomsHandler : IExternalEventHandler
{
    private readonly RoomInspectorViewModel _vm;

    public LoadRoomsHandler(RoomInspectorViewModel vm) => _vm = vm;
    public string GetName() => "Load Rooms";

    public void Execute(UIApplication uiapp)
    {
        Document doc = uiapp.ActiveUIDocument?.Document;
        if (doc == null) return;

        // Query Revit on the main thread
        var rooms = new FilteredElementCollector(doc)
            .OfCategory(BuiltInCategory.OST_Rooms)
            .Cast<Room>()
            .Where(r => r.Area > 0)
            .Select(r => new RoomData
            {
                RevitId    = r.Id,
                Name       = r.Name,
                Number     = r.Number,
                Department = r.get_Parameter(
                    BuiltInParameter.ROOM_DEPARTMENT)?.AsString() ?? "",
                AreaSqM    = r.Area * 0.092903,
                Level      = r.Level?.Name ?? ""
            })
            .OrderBy(r => r.Number)
            .ToList();

        // Update ViewModel on the UI dispatcher thread
        System.Windows.Application.Current?.Dispatcher.Invoke(() =>
            _vm.UpdateRooms(rooms));
    }
}

// ── SelectElementHandler.cs ──
public class SelectElementHandler : IExternalEventHandler
{
    public ElementId ElementId { get; set; }
    public string GetName() => "Select Element";
    public void Execute(UIApplication uiapp)
    {
        if (ElementId == null) return;
        var uidoc = uiapp.ActiveUIDocument;
        uidoc?.Selection.SetElementIds(
            new List<ElementId> { ElementId });
        uidoc?.ShowElements(ElementId);
    }
}

// ── Application.cs — wire everything up ──
public class StudioApplication : IExternalApplication
{
    public static ExternalEvent         LoadRoomsEvent;
    public static ExternalEvent         SelectElementEvent;
    public static SelectElementHandler  SelectElementHandler;
    public static RoomInspectorViewModel ViewModel;

    public Result OnStartup(UIControlledApplication app)
    {
        ViewModel = new RoomInspectorViewModel();

        // Create events BEFORE registering pane
        LoadRoomsEvent     = ExternalEvent.Create(new LoadRoomsHandler(ViewModel));
        SelectElementHandler = new SelectElementHandler();
        SelectElementEvent   = ExternalEvent.Create(SelectElementHandler);

        // Register pane (WPF UserControl uses ViewModel)
        var paneId = new DockablePaneId(new Guid("AABB1122-3344-5566-ROOM-INSP00000001"));
        var pane   = new RoomInspectorPane(ViewModel);
        app.RegisterDockablePane(paneId, "Room Inspector", pane);

        return Result.Succeeded;
    }

    public Result OnShutdown(UIControlledApplication app)
    {
        LoadRoomsEvent?.Dispose();
        SelectElementEvent?.Dispose();
        return Result.Succeeded;
    }
}

SetupRevitTestFramework Project Setup

using NUnit.Framework;
using Autodesk.Revit.DB;
using RevitTestFramework;

[TestFixture]
public class WallCountTests
{
    // TestModel attribute loads a specific .rvt file for this test
    [Test]
    [TestModel(@"TestModels\SampleBuilding.rvt")]
    public void WallCount_SampleBuilding_Returns47Walls()
    {
        // RTF injects the Document via RevitTestFramework.RevitTestBase
        Document doc = RevitTestBase.CurrentDocument;
        Assert.IsNotNull(doc);

        int wallCount = new FilteredElementCollector(doc)
            .OfClass(typeof(Wall)).GetElementCount();

        Assert.AreEqual(47, wallCount,
            "Expected 47 walls in sample building");
    }

    [Test]
    [TestModel(@"TestModels\SampleBuilding.rvt")]
    public void ReinforcBeam_CreatesCorrectBarCount()
    {
        Document doc = RevitTestBase.CurrentDocument;

        // Get the first beam in the test model
        FamilyInstance beam = new FilteredElementCollector(doc)
            .OfCategory(BuiltInCategory.OST_StructuralFraming)
            .WhereElementIsNotElementType()
            .Cast<FamilyInstance>()
            .FirstOrDefault(fi => fi.StructuralType == StructuralType.Beam);

        Assert.IsNotNull(beam, "No beam found in test model");

        int rebarBefore = new FilteredElementCollector(doc)
            .OfCategory(BuiltInCategory.OST_Rebar)
            .GetElementCount();

        // Run the reinforcement command
        var cmd = new StudioTools.Commands.ReinforcBeamCommand();
        cmd.ReinforceSingleBeam(doc, beam);

        int rebarAfter = new FilteredElementCollector(doc)
            .OfCategory(BuiltInCategory.OST_Rebar)
            .GetElementCount();

        // Expect: 3 bottom + 2 top + ~10 stirrups = 15 bars minimum
        Assert.Greater(rebarAfter - rebarBefore, 14,
            "Expected at least 15 rebar elements after reinforcing beam");
    }

    [Test]
    [TestModel(@"TestModels\Empty.rvt")]
    public void CreateWall_ValidLine_ReturnsWallElement()
    {
        Document doc = RevitTestBase.CurrentDocument;

        Level level = new FilteredElementCollector(doc)
            .OfClass(typeof(Level)).Cast<Level>().First();

        Wall wall = null;
        using (var tx = new Transaction(doc, "Test Wall"))
        {
            tx.Start();
            wall = Wall.Create(doc,
                Line.CreateBound(XYZ.Zero, new XYZ(10,0,0)),
                level.Id, false);
            tx.Commit();
        }

        Assert.IsNotNull(wall);
        Assert.AreEqual(ElementId.InvalidElementId, wall.LevelId == null
            ? ElementId.InvalidElementId : level.Id);
    }
}

PatternAuto-Update Check on Startup
🟢 Check GitHub releases for new version on startup
// Check latest version from GitHub API + prompt user to update
public static class UpdateChecker
{
    private const string CURRENT_VERSION    = "2.1.0";
    private const string GITHUB_RELEASES_URL =
        "https://api.github.com/repos/yourstudio/studiotools/releases/latest";

    /// Check for updates asynchronously — called from ApplicationInitialized event
    public static async void CheckAsync()
    {
        try
        {
            using var client = new System.Net.Http.HttpClient();
            client.DefaultRequestHeaders.UserAgent.ParseAdd("StudioTools/2.1.0");

            string json = await client.GetStringAsync(GITHUB_RELEASES_URL);
            dynamic release = Newtonsoft.Json.JsonConvert
                .DeserializeObject(json);

            string latestTag = (string)release.tag_name;  // e.g. "v2.2.0"
            string latestVer = latestTag.TrimStart('v');

            if (new Version(latestVer) > new Version(CURRENT_VERSION))
            {
                string notes = (string)release.body ?? "";
                string url   = (string)release.html_url;

                // Show on UI thread
                System.Windows.Application.Current?.Dispatcher.Invoke(() =>
                {
                    var result = System.Windows.MessageBox.Show(
                        $"Studio Tools v{latestVer} is available!\n\n" +
                        $"You have v{CURRENT_VERSION}.\n\n" +
                        $"Changes:\n{notes}\n\n" +
                        "Download now?",
                        "Update Available",
                        System.Windows.MessageBoxButton.YesNo,
                        System.Windows.MessageBoxImage.Information);

                    if (result == System.Windows.MessageBoxResult.Yes)
                        System.Diagnostics.Process.Start(url);
                });
            }
        }
        catch (Exception ex)

using System.Net.Http;
using System.Net.Http.Headers;
using Newtonsoft.Json;

// ── Singleton HttpClient (reuse — never new up per request) ──
public static class ApiService
{
    private static readonly HttpClient _client;
    private const string BASE_URL = "https://api.yoursystem.com/v1";

    static ApiService()
    {
        _client = new HttpClient();
        _client.BaseAddress = new Uri(BASE_URL);
        _client.DefaultRequestHeaders.Accept.Add(
            new MediaTypeWithQualityHeaderValue("application/json"));
        _client.Timeout = TimeSpan.FromSeconds(30);
    }

    // ── POST room data to external facilities management system ──
    public static async Task<bool> PostRoomsAsync(
        IList<RoomData> rooms, string apiKey)
    {
        try
        {
            _client.DefaultRequestHeaders.Authorization =
                new AuthenticationHeaderValue("Bearer", apiKey);

            var payload = new
            {
                project  = "Building A",
                exported = DateTime.UtcNow,
                rooms    = rooms.Select(r => new
                {
                    id         = r.RevitId?.IntegerValue,
                    name       = r.Name,
                    number     = r.Number,
                    area_sqm   = r.AreaSqM,
                    level      = r.Level,
                    department = r.Department
                })
            };

            string json    = JsonConvert.SerializeObject(payload);
            var    content  = new StringContent(json, Encoding.UTF8, "application/json");
            HttpResponseMessage resp = await _client.PostAsync("/rooms", content);

            resp.EnsureSuccessStatusCode();
            PluginLogger.Log.Information(
                "Posted {Count} rooms to API. Status: {Status}",
                rooms.Count, resp.StatusCode);
            return true;
        }
        catch (Exception ex)
        {
            PluginLogger.Log.Error(ex, "Failed to POST rooms to API");
            return false;
        }
    }

    // ── GET config from API ──
    public static async Task<PluginConfig> GetConfigAsync()
    {
        try
        {
            string json = await _client.GetStringAsync("/config/studiotools");
            return JsonConvert.DeserializeObject<PluginConfig>(json);
        }
        catch (Exception ex)
        {
            PluginLogger.Log.Warning(ex, "Could not fetch config from API, using defaults");
            return new PluginConfig();  // return defaults
        }
    }
}

// LocalDatabase.cs — SQLite local cache
using Microsoft.Data.Sqlite;

public class LocalDatabase
{
    private readonly string _connectionString;

    public LocalDatabase()
    {
        string dbDir = System.IO.Path.Combine(
            System.Environment.GetFolderPath(
                System.Environment.SpecialFolder.ApplicationData),
            "StudioTools");
        System.IO.Directory.CreateDirectory(dbDir);

        string dbPath = System.IO.Path.Combine(dbDir, "studiotools.db");
        _connectionString = $"Data Source={dbPath}";
        InitialiseSchema();
    }

    private void InitialiseSchema()
    {
        using var conn = new SqliteConnection(_connectionString);
        conn.Open();

        // Create tables if they don't exist
        using var cmd = conn.CreateCommand();
        cmd.CommandText = @"
            CREATE TABLE IF NOT EXISTS ExportHistory (
                Id          INTEGER PRIMARY KEY AUTOINCREMENT,
                ProjectName TEXT    NOT NULL,
                ExportedAt  TEXT    NOT NULL,
                ItemCount   INTEGER NOT NULL,
                ExportType  TEXT    NOT NULL
            );

            CREATE TABLE IF NOT EXISTS RoomCache (
                RevitId     INTEGER PRIMARY KEY,
                ProjectPath TEXT    NOT NULL,
                Name        TEXT,
                Number      TEXT,
                AreaSqM     REAL,
                Department  TEXT,
                CachedAt    TEXT    NOT NULL
            );";
        cmd.ExecuteNonQuery();
    }

    // ── INSERT export history record ──
    public void LogExport(string project, int count, string type)
    {
        using var conn = new SqliteConnection(_connectionString);
        conn.Open();
        using var cmd = conn.CreateCommand();
        cmd.CommandText = 
            "INSERT INTO ExportHistory (ProjectName, ExportedAt, ItemCount, ExportType)
             VALUES (@p, @d, @c, @t)";
        cmd.Parameters.AddWithValue("@p", project);
        cmd.Parameters.AddWithValue("@d", DateTime.UtcNow.ToString("o"));
        cmd.Parameters.AddWithValue("@c", count);
        cmd.Parameters.AddWithValue("@t", type);
        cmd.ExecuteNonQuery();
    }

    // ── SELECT recent export history ──
    public List<(string Project, DateTime When, int Count, string Type)>
        GetRecentExports(int limit = 20)
    {
        var result = new List<(string, DateTime, int, string)>();
        using var conn = new SqliteConnection(_connectionString);
        conn.Open();
        using var cmd = conn.CreateCommand();
        cmd.CommandText = 
            "SELECT ProjectName, ExportedAt, ItemCount, ExportType
             FROM ExportHistory ORDER BY Id DESC LIMIT @limit";
        cmd.Parameters.AddWithValue("@limit", limit);

        using var reader = cmd.ExecuteReader();
        while (reader.Read())
            result.Add((
                reader.GetString(0),
                DateTime.Parse(reader.GetString(1)),
                reader.GetInt32(2),
                reader.GetString(3)));

        return result;
    }

    // ── Cache rooms from a project ──
    public void CacheRooms(string projectPath,
        IEnumerable<RoomData> rooms)
    {
        using var conn = new SqliteConnection(_connectionString);
        conn.Open();
        using var txn = conn.BeginTransaction();

        foreach (var r in rooms)
        {
            using var cmd = conn.CreateCommand();
            cmd.Transaction = txn;
            cmd.CommandText = @"
                INSERT OR REPLACE INTO RoomCache
                    (RevitId,ProjectPath,Name,Number,AreaSqM,Department,CachedAt)
                VALUES (@id,@pp,@n,@nu,@a,@d,@ca)";
            cmd.Parameters.AddWithValue("@id", r.RevitId?.IntegerValue ?? 0);
            cmd.Parameters.AddWithValue("@pp", projectPath);
            cmd.Parameters.AddWithValue("@n",  r.Name ?? "");
            cmd.Parameters.AddWithValue("@nu", r.Number ?? "");
            cmd.Parameters.AddWithValue("@a",  r.AreaSqM);
            cmd.Parameters.AddWithValue("@d",  r.Department ?? "");
            cmd.Parameters.AddWithValue("@ca", DateTime.UtcNow.ToString("o"));
            cmd.ExecuteNonQuery();
        }
        txn.Commit();
    }
}

using ClosedXML.Excel;

public static class ExcelService
{
    // ── READ: Import room data from Excel template ──
    public static List<RoomData> ReadRoomSchedule(string xlsxPath)
    {
        var result = new List<RoomData>();

        using var wb = new XLWorkbook(xlsxPath);
        var ws = wb.Worksheet(1);  // first sheet

        // Find header row — look for "Room Number" header
        int startRow = 2;  // row 1 = headers
        int lastRow  = ws.LastRowUsed()?.RowNumber() ?? 1;

        for (int row = startRow; row <= lastRow; row++)
        {
            string number = ws.Cell(row, 1).GetString();
            if (string.IsNullOrEmpty(number)) continue;

            result.Add(new RoomData
            {
                Number     = number,
                Name       = ws.Cell(row, 2).GetString(),
                Department = ws.Cell(row, 3).GetString(),
                AreaSqM    = ws.Cell(row, 4).GetDouble()
            });
        }

        System.Diagnostics.Debug.WriteLine(
            $"Read {result.Count} rooms from {xlsxPath}");
        return result;
    }

    // ── WRITE: Export Revit rooms to formatted Excel ──
    public static void ExportRoomSchedule(
        IEnumerable<RoomData> rooms, string outputPath)
    {
        using var wb = new XLWorkbook();
        var ws = wb.Worksheets.Add("Room Schedule");

        // ── Header row ──
        string[] headers = { "Room No.", "Room Name", "Department",
                              "Level", "Area (m²)", "Revit ID" };
        for (int c = 0; c < headers.Length; c++)
        {
            var cell = ws.Cell(1, c + 1);
            cell.Value = headers[c];
            cell.Style.Font.Bold = true;
            cell.Style.Fill.SetBackgroundColor(XLColor.FromArgb(68, 114, 196));
            cell.Style.Font.SetFontColor(XLColor.White);
        }

        // ── Data rows ──
        int row = 2;
        foreach (var r in rooms)
        {
            ws.Cell(row, 1).Value = r.Number;
            ws.Cell(row, 2).Value = r.Name;
            ws.Cell(row, 3).Value = r.Department;
            ws.Cell(row, 4).Value = r.Level;
            ws.Cell(row, 5).Value = r.AreaSqM;
            ws.Cell(row, 5).Style.NumberFormat.Format = "0.00";
            ws.Cell(row, 6).Value = r.RevitId?.IntegerValue ?? 0;

            // Alternate row colour
            if (row % 2 == 0)
                ws.Row(row).Style.Fill.SetBackgroundColor(
                    XLColor.FromArgb(242, 242, 242));
            row++;
        }

        // ── Auto-fit columns + freeze header row ──
        ws.Columns().AdjustToContents();
        ws.SheetView.FreezeRows(1);

        // ── Add total row ──
        ws.Cell(row, 4).Value = "TOTAL";
        ws.Cell(row, 4).Style.Font.Bold = true;
        ws.Cell(row, 5).FormulaA1 = $"=SUM(E2:E{row-1})";
        ws.Cell(row, 5).Style.Font.Bold = true;

        wb.SaveAs(outputPath);
        System.Diagnostics.Debug.WriteLine(
            $"Exported {row-2} rooms to {outputPath}");
    }

public class BatchReinforcementHandler : IExternalEventHandler
{
    public IList<ElementId>   BeamIds      { get; set; }
    public ProgressViewModel ProgressVm   { get; set; }

    public string GetName() => "Batch Reinforce";

    public void Execute(UIApplication uiapp)
    {
        Document doc     = uiapp.ActiveUIDocument?.Document;
        var      ct      = ProgressVm.CancelSource.Token;
        int      total   = BeamIds.Count;
        int      done    = 0;

        // Update progress on UI thread
        void Report(string msg)
        {
            System.Windows.Application.Current?.Dispatcher.Invoke(() =>
            {
                ProgressVm.Value   = done;
                ProgressVm.Maximum = total;
                ProgressVm.Message = msg;
                OnPropertyChanged(nameof(ProgressVm.PercentText));
            });
        }

        Report("Starting reinforcement...");

        foreach (var id in BeamIds)
        {
            // Check cancellation before each beam
            if (ct.IsCancellationRequested) break;

            FamilyInstance beam = doc?.GetElement(id) as FamilyInstance;
            if (beam == null) continue;

            try
            {
                Report($"Reinforcing beam {done+1} of {total}: {beam.Name}");
                // Do the reinforcement (Book 17 methods)
                ReinforceSingleBeam(doc, beam);
            }
            catch (Exception ex)
            {
                PluginLogger.Log.Warning(ex,
                    "Failed to reinforce beam {Id}", id);
            }
            finally { done++; }
        }

        System.Windows.Application.Current?.Dispatcher.Invoke(() =>
        {
            ProgressVm.Value   = done;
            ProgressVm.Message = ProgressVm.IsCancelled
                ? $"Cancelled after {done} beams"
                : $"Complete — {done} beams reinforced";
        });
    }
}

PatternLicense Key Validation
Simple HMAC-based license key validation — no external server needed for offline license checks. For commercial plugins: generate keys tied to machine ID + expiry date, validate on startup.
🟢 Machine-locked license validation
// LicenseManager.cs — offline HMAC license validation
public static class LicenseManager
{
    private const string SECRET = "your-secret-key-change-this-in-production";

    // Get unique machine fingerprint
    public static string GetMachineId()
    {
        string cpu = System.Environment.GetEnvironmentVariable("PROCESSOR_IDENTIFIER")
                  ?? System.Environment.MachineName;
        string user = System.Environment.UserName;
        string raw  = $"{cpu}|{user}";
        using var sha = System.Security.Cryptography.SHA256.Create();
        var hash = sha.ComputeHash(Encoding.UTF8.GetBytes(raw));
        return Convert.ToBase64String(hash)[..16];
    }

    // Generate a license key (run server-side when customer buys)
    public static string GenerateKey(string machineId, DateTime expiry)
    {
        string data = $"{machineId}|{expiry:yyyyMMdd}";
        using var hmac = new System.Security.Cryptography.HMACSHA256(
            Encoding.UTF8.GetBytes(SECRET));
        var hash = hmac.ComputeHash(Encoding.UTF8.GetBytes(data));
        string sig = Convert.ToBase64String(hash)[..16];
        return $"{machineId}|{expiry:yyyyMMdd}|{sig}";
    }

    // Validate a license key (run on every startup)
    public static (bool IsValid, string Reason, DateTime Expiry) Validate(string licenseKey)
    {
        if (string.IsNullOrWhiteSpace(licenseKey))
            return (false, "No license key provided", DateTime.MinValue);

        try
        {
            string[] parts = licenseKey.Split('|');
            if (parts.Length != 3)
                return (false, "Invalid key format", DateTime.MinValue);

            string keyMachineId = parts[0];
            string expiryStr    = parts[1];
            string signature    = parts[2];

            // Check machine lock
            string myMachineId = GetMachineId();
            if (keyMachineId != myMachineId)
                return (false, "License is not valid for this machine", DateTime.MinValue);

            // Verify HMAC signature
            string data = $"{keyMachineId}|{expiryStr}";
            using var hmac = new System.Security.Cryptography.HMACSHA256(
                Encoding.UTF8.GetBytes(SECRET));
            var    hash       = hmac.ComputeHash(Encoding.UTF8.GetBytes(data));
            string expectedSig = Convert.ToBase64String(hash)[..16];
            if (signature != expectedSig)
                return (false, "Invalid license signature", DateTime.MinValue);

            // Check expiry
            DateTime expiry = DateTime.ParseExact(
                expiryStr, "yyyyMMdd", null);
            if (DateTime.Today > expiry)
                return (false,
                    $"License expired on {expiry:dd MMM yyyy}", expiry);

            return (true,
                $"Valid until {expiry:dd MMM yyyy}", expiry);
        }
        catch (Exception ex)
        {
            return (false, $"License error: {ex.Message}", DateTime.MinValue);
        }
    }
}

// ── Check on startup ──
public Result OnStartup(UIControlledApplication app)
{
    var (isValid, reason, expiry) = LicenseManager.Validate(
        PluginSettings.Current.LicenseKey);

    if (!isValid)
    {
        TaskDialog.Show("Studio Tools — License",
            $"License validation failed:\n{reason}\n\n" +
            "Please contact support@yourstudio.com");
        // Return Succeeded but disable commands via IExternalCommandAvailability
        MyApp.IsLicensed = false;
    }
    else
    {
        MyApp.IsLicensed = true;
        // Warn if expiring within 30 days
        if ((expiry - DateTime.Today).TotalDays < 30)
            TaskDialog.Show("License Expiry",
                $"Your license expires in {(expiry-DateTime.Today).Days} days.\n" +
                "Please renew at https://yourstudio.com/renew");
    }
    return Result.Succeeded;
}
