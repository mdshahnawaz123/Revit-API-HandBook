C# — Collect all views
// All views (excluding templates)
var allViews = new FilteredElementCollector(doc)
    .OfClass(typeof(View))
    .Cast<View>()
    .Where(v => !v.IsTemplate)
    .ToList();

foreach (var view in allViews)
{
    string   name     = view.Name;
    ViewType type     = view.ViewType;
    int      scale    = view.Scale;       // e.g. 100 for 1:100
    bool     isTmpl  = view.IsTemplate;
    ElementId tmplId = view.ViewTemplateId;
}
// 02

Create Floor Plan Views

// Critical Use ViewFamilyType ID (not a ViewPlan ID) in ViewPlan.Create(). These are completely different types.
C#
// Get the ViewFamilyType for FloorPlan
ElementId floorPlanTypeId = new FilteredElementCollector(doc)
    .OfClass(typeof(ViewFamilyType))
    .Cast<ViewFamilyType>()
    .FirstOrDefault(x => x.ViewFamily == ViewFamily.FloorPlan)?.Id;

var levels = new FilteredElementCollector(doc)
    .OfClass(typeof(Level)).Cast<Level>().ToList();

using (var tx = new Transaction(doc, "Create Floor Plans"))
{
    tx.Start();
    foreach (var lvl in levels)
    {
        var name = $"Auto_FloorPlan_{lvl.Name}";

        bool exists = new FilteredElementCollector(doc)
            .OfClass(typeof(ViewPlan)).Any(v => v.Name == name);
        if (exists) continue;

        ViewPlan vp = ViewPlan.Create(doc, floorPlanTypeId, lvl.Id);
        vp.Name = name;
        vp.Scale = 100;
    }
    tx.Commit();
}
// 03

Create Section Views

Sections require a BoundingBoxXYZ with a Transform that defines position and orientation.

C#
ElementId sectionTypeId = new FilteredElementCollector(doc)
    .OfClass(typeof(ViewFamilyType))
    .Cast<ViewFamilyType>()
    .FirstOrDefault(x => x.ViewFamily == ViewFamily.Section)?.Id;

// For a wall cross-section:
LocationCurve lc       = wall.Location as LocationCurve;
XYZ           midPt    = lc.Curve.Evaluate(0.5, true);
XYZ           wallDir  = (lc.Curve.GetEndPoint(1) - lc.Curve.GetEndPoint(0)).Normalize();
XYZ           upDir    = XYZ.BasisZ;
XYZ           viewDir  = wallDir.CrossProduct(upDir).Normalize();

Transform t = Transform.Identity;
t.Origin  = midPt;
t.BasisX  = wallDir;   // horizontal in view
t.BasisY  = upDir;     // vertical in view
t.BasisZ  = viewDir;   // camera direction (into wall)

BoundingBoxXYZ bb = new BoundingBoxXYZ
{
    Transform = t,
    Min = new XYZ(-10, 0, -5),
    Max = new XYZ( 10, 10,  5)
};

using (var tx = new Transaction(doc, "Create Section"))
{
    tx.Start();
    ViewSection section = ViewSection.CreateSection(doc, sectionTypeId, bb);
    section.Name = $"Section_{wall.Id}";
    tx.Commit();
}

Rotate a Section View 90°

C#
using (var tx = new Transaction(doc, "Rotate Section"))
{
    tx.Start();
    BoundingBoxXYZ bb  = section.GetCropBox();
    XYZ            oldX = bb.Transform.BasisX;
    XYZ            oldY = bb.Transform.BasisY;
    bb.Transform.BasisX = -oldY;    // 90° rotation
    bb.Transform.BasisY =  oldX;
    section.SetCropBox(bb);
    section.CropBoxActive = true;
    tx.Commit();
}

// 05

View Filters & Graphic Overrides

C#
// 1. Build the parameter filter rule (exterior walls = function 0)
var categories = new List<ElementId>
    { new ElementId(BuiltInCategory.OST_Walls) };

var provider  = new ParameterValueProvider(
    new ElementId(BuiltInParameter.FUNCTION_PARAM));
var rule      = new FilterDoubleRule(
    provider, new FilterNumericEquals(), 0, 1e-6);
var elemFilter = new ElementParameterFilter(rule);

using (var tx = new Transaction(doc, "Add View Filter"))
{
    tx.Start();

    // 2. Create the ParameterFilterElement (check for duplicates first)
    bool exists = new FilteredElementCollector(doc)
        .OfClass(typeof(ParameterFilterElement))
        .Any(f => f.Name == "Exterior Walls");

    ParameterFilterElement pfe = exists
        ? new FilteredElementCollector(doc)
            .OfClass(typeof(ParameterFilterElement))
            .Cast<ParameterFilterElement>()
            .First(f => f.Name == "Exterior Walls")
        : ParameterFilterElement.Create(
            doc, "Exterior Walls", categories, elemFilter);

    // 3. Build color override
    var overrides = new OverrideGraphicSettings();
    overrides.SetCutFillColor(new Color(255, 0, 0));
    overrides.SetProjectionLineColor(new Color(255, 0, 0));
    overrides.SetSurfaceTransparency(50);

    // 4. Apply to all floor plan views
    var plans = new FilteredElementCollector(doc)
        .OfClass(typeof(ViewPlan)).Cast<ViewPlan>()
        .Where(v => !v.IsTemplate).ToList();

    foreach (var view in plans)
    {
        view.AddFilter(pfe.Id);
        view.SetFilterOverrides(pfe.Id, overrides);
        view.SetFilterVisibility(pfe.Id, true);
    }

    tx.Commit();
}

Dimensions

See Book 3 Supplement for the complete dimensions handbook. Here is the core pattern.

C#
// Always: View + ReferenceArray + Line → NewDimension()
ReferenceArray refs = new ReferenceArray();
refs.Append(curve.GetEndPointReference(0));   // start
refs.Append(curve.GetEndPointReference(1));   // end

Line dimLine = Line.CreateBound(
    curve.GetEndPoint(0) + new XYZ(0,3,0),
    curve.GetEndPoint(1) + new XYZ(0,3,0));

using (var tx = new Transaction(doc, "Dimension"))
{
    tx.Start();
    Dimension dim = doc.Create.NewDimension(doc.ActiveView, dimLine, refs);
    tx.Commit();

    // Read value (internal feet → mm)
    if (dim.Value.HasValue)
    {
        double mm = dim.Value.Value * 304.8;
    }
}

// Reference types summary:
// Grids:        new Reference(grid)                ← CORRECT
// Wall faces:   face.Reference (ComputeReferences=true)
// Endpoints:    curve.GetEndPointReference(0 or 1)
// Families:     fi.GetReferences(FamilyInstanceReferenceType.xxx)

Text Notes & Tags

// Get a TextNoteType
TextNoteType tnt = new FilteredElementCollector(doc)
    .OfClass(typeof(TextNoteType))
    .Cast<TextNoteType>()
    .First();

using (var tx = new Transaction(doc, "Add Text"))
{
    tx.Start();

    // Create text note
    TextNote note = TextNote.Create(doc, doc.ActiveView.Id,
        new XYZ(5, 5, 0),    // position
        "Hello Revit API",  // text
        tnt.Id);            // type

    // Tag an element (IndependentTag)
    IndependentTag tag = IndependentTag.Create(
        doc,
        doc.ActiveView.Id,
        new Reference(wall),
        false,               // hasLeader
        TagMode.TM_ADDBY_CATEGORY,
        TagOrientation.Horizontal,
        new XYZ(5, 5, 0));  // tag head position

    tx.Commit();
}

Detail Lines & Filled Regions

C#
// Get a line style
Category   lineCat    = doc.Settings.Categories.get_Item(BuiltInCategory.OST_Lines);
GraphicsStyle lineStyle = lineCat.SubCategories.get_Item("Wide Lines")
    .GetGraphicsStyle(GraphicsStyleType.Projection);

using (var tx = new Transaction(doc, "Detail Line"))
{
    tx.Start();

    // Detail line in active view
    DetailLine dl = doc.Create.NewDetailCurve(
        doc.ActiveView,
        Line.CreateBound(new XYZ(0,0,0), new XYZ(5,0,0)))
        as DetailLine;
    dl.LineStyle = lineStyle;

    // Filled region
    FilledRegionType frt = new FilteredElementCollector(doc)
        .OfClass(typeof(FilledRegionType)).Cast<FilledRegionType>().First();

    var loop = new CurveLoop();
    // ... add curves to loop ...

    FilledRegion fr = FilledRegion.Create(
        doc, frt.Id, doc.ActiveView.Id,
        new List<CurveLoop> { loop });

    tx.Commit();
