// Family API

Family · FamilyManager · DirectShape

Create, edit, and manage Revit families programmatically — types, parameters, geometry.

ClassFamily
Inherits: Element
Namespace: Autodesk.Revit.DB · Assembly: RevitAPI.dll
Represents a loaded family in the project. A Family is the container for one or more FamilySymbol types. It provides access to all symbols, indicates whether it's an in-place family, and its category. Use EditFamily() to open the family for editing, or LoadFamily() to load it back after changes.
🟠 Key Properties
🟠public bool IsInPlace { get; }
True if this is an in-place family (created within the project rather than loaded from a .rfa file). In-place families cannot be directly converted to loadable families through the API.
🟠public bool IsEditable { get; }
True if the family document can be opened for editing. System families (walls, floors etc.) are not editable.
🟠public Category FamilyCategory { get; set; }
Gets or sets the category of this family (Doors, Windows, Furniture, etc.).
🟢 Key Methods
🟢public ISet<ElementId> GetFamilySymbolIds()
Returns the ElementIds of all FamilySymbol types within this family. Iterate and cast each to FamilySymbol to access type parameters.
Returns: ISet<ElementId> — Set of FamilySymbol IDs.
// Example — Family inventory and editing workflow
// ── Inventory all families ──
var families = new FilteredElementCollector(doc)
    .OfClass(typeof(Family))
    .Cast<Family>()
    .OrderBy(f => f.Name)
    .ToList();

foreach (var fam in families)
{
    int    typeCount = fam.GetFamilySymbolIds().Count;
    bool   inPlace  = fam.IsInPlace;
    string cat      = fam.FamilyCategory?.Name ?? "Unknown";

    // Count placed instances of this family
    int instances = new FilteredElementCollector(doc)
        .OfClass(typeof(FamilyInstance))
        .Cast<FamilyInstance>()
        .Count(fi => fi.Symbol?.Family?.Id == fam.Id);

    System.Diagnostics.Debug.WriteLine(
        $"{fam.Name} [{cat}] Types:{typeCount} Instances:{instances} InPlace:{inPlace}");
}

// ── Load a .rfa file ──
using (var tx = new Transaction(doc, "Load Family"))
{
    tx.Start();
    Family loaded;
    bool ok = doc.LoadFamily(
        @"C:\Revit\Families\SingleFlushDoor.rfa",
        new FamilyLoadOverwriteOptions(),
        out loaded);
    tx.Commit();
}

// ── Open family for editing ──
Family doorFamily = families.First(f => f.Name.Contains("Door"));
if (doorFamily.IsEditable)
{
    Document famDoc = doc.EditFamily(doorFamily);
    // ... edit famDoc ...
    famDoc.LoadFamily(doc, new FamilyLoadOverwriteOptions());
    famDoc.Close(false);
}
ClassFamilyManager
Inherits: APIObject
Namespace: Autodesk.Revit.DB · only in family documents
Manages family types and parameters within an open family document. Access via famDoc.FamilyManager. Lets you create/delete/rename types, add/remove/modify parameters, set parameter values, and define formulas. Only available when the active document is a .rfa family document.
// Family Document OnlyFamilyManager is only valid inside a family document (doc.IsFamilyDocument == true). Calling it on a project document throws an exception. Always check before use.
🟠 Key Properties
🟠public FamilyType CurrentType { get; set; }
Gets or sets the currently active type. Parameter values read/written via FamilyManager apply to this type. Set this before calling Get/Set to target a specific type.
🟠public FamilyTypeSet Types { get; }
Gets all FamilyType objects in this family. Enumerate to iterate over all types.
🟠public FamilyParameterSet Parameters { get; }
Gets all FamilyParameter definitions in this family.
🟢 Key Methods
🟢public FamilyType NewType(string name)
Creates a new family type with the given name. The new type becomes CurrentType. Parameter values are copied from the previous current type. Must be inside a Transaction.
🟢public FamilyParameter AddParameter(string parameterName, ForgeTypeId groupTypeId, ForgeTypeId specTypeId, bool isInstance)
Adds a new family parameter. groupTypeId controls which group it appears under in properties (GroupTypeId.Dimensions, GroupTypeId.Data, etc.). specTypeId is the data type (SpecTypeId.Length, SpecTypeId.Boolean, SpecTypeId.String, etc.).
🟢public void Set(FamilyParameter familyParameter, double value)
Sets a parameter value for the CurrentType. Also has overloads for Set(param, int), Set(param, string), Set(param, ElementId). Value must be in internal units for double parameters.
🟢public void SetFormula(FamilyParameter familyParameter, string formula)
Sets a parameter formula (e.g. "Width * 2"). The formula string uses other parameter names and arithmetic operators. Pass null or empty string to remove the formula.
// Example — Complete FamilyManager workflow
// Open a family document
Document famDoc = doc.EditFamily(doorFamily);
FamilyManager mgr = famDoc.FamilyManager;

using (var tx = new Transaction(famDoc, "Edit Family"))
{
    tx.Start();

    // Add a new Length parameter
    FamilyParameter frameWidth = mgr.AddParameter(
        "FrameWidth",
        GroupTypeId.Dimensions,
        SpecTypeId.Length,
        false);  // false = type parameter

    // Add a Yes/No instance parameter
    FamilyParameter hasGlass = mgr.AddParameter(
        "HasGlassPanel",
        GroupTypeId.General,
        SpecTypeId.Boolean.YesNo,
        true);   // true = instance parameter

    // Set formula: FrameWidth = Width / 10
    FamilyParameter width = mgr.Parameters.Cast<FamilyParameter>()
        .FirstOrDefault(p => p.Definition.Name == "Width");
    if (width != null)
        mgr.SetFormula(frameWidth, "Width / 10");

    // Create new types from a data table
    var typeSizes = new (string name, double widthMm, double heightMm)[]
    {
        ("900x2100", 900, 2100),
        ("1000x2100", 1000, 2100),
        ("1200x2400", 1200, 2400),
    };

    foreach (var (name, wMm, hMm) in typeSizes)
    {
        mgr.NewType(name);   // creates and makes current

        var wParam = mgr.Parameters.Cast<FamilyParameter>()
            .First(p => p.Definition.Name == "Width");
        var hParam = mgr.Parameters.Cast<FamilyParameter>()
            .First(p => p.Definition.Name == "Height");

        mgr.Set(wParam, UnitUtils.ConvertToInternalUnits(wMm, UnitTypeId.Millimeters));
        mgr.Set(hParam, UnitUtils.ConvertToInternalUnits(hMm, UnitTypeId.Millimeters));
    }

    // List all types with their parameter values
    foreach (FamilyType ft in mgr.Types)
    {
        mgr.CurrentType = ft;
        System.Diagnostics.Debug.WriteLine($"Type: {ft.Name}");
        foreach (FamilyParameter fp in mgr.Parameters)
        {
            Parameter val = mgr.GetValue(fp);
            System.Diagnostics.Debug.WriteLine(
                $"  {fp.Definition.Name} = {val?.AsValueString()}");
        }
    }

    tx.Commit();
}

// Load modified family back into project
famDoc.LoadFamily(doc, new FamilyLoadOverwriteOptions());
famDoc.Close(false);
InterfaceIFamilyLoadOptions
Namespace: Autodesk.Revit.DB
Controls behaviour when loading a family that already exists in the project. Implement this interface to silently overwrite existing families without showing a confirmation dialog.
🟢 Implementation Pattern
// Example — Silent overwrite implementation
// Implement IFamilyLoadOptions to always overwrite
public class FamilyLoadOverwriteOptions : IFamilyLoadOptions
{
    // Called when a family with the same name already exists
    public bool OnFamilyFound(bool familyInUse,
        out bool overwriteParamValues)
    {
        overwriteParamValues = true;  // overwrite parameter values too
        return true;                  // true = proceed with load
    }

    // Called when a shared family is found
    public bool OnSharedFamilyFound(Family sharedFamily,
        bool familyInUse,
        out FamilySource source,
        out bool overwriteParamValues)
    {
        source = FamilySource.Family;  // use the file version
        overwriteParamValues = true;
        return true;
    }
}

// Usage
using (var tx = new Transaction(doc, "Reload Family"))
{
    tx.Start();
    Family result;
    doc.LoadFamily(path, new FamilyLoadOverwriteOptions(), out result);
    tx.Commit();
}
ClassDirectShape
Inherits: Element
Namespace: Autodesk.Revit.DB
A lightweight element that displays custom geometry without requiring a full family document. DirectShapes are created from Solid, Mesh, or TessellatedShapeBuilder geometry. They're ideal for visualisation, IFC import, analysis overlays, and temporary geometry. DirectShapes appear in views but cannot host other elements.
🟢 Static Factory + Usage
🟣public static DirectShape CreateElement(Document doc, ElementId categoryId)
Creates an empty DirectShape element assigned to the given category. After creation, call SetShape() to assign geometry. The categoryId determines how the shape appears in schedules and visibility filters.
// Example — DirectShape from solid + from tessellated builder
using (var tx = new Transaction(doc, "DirectShape"))
{
    tx.Start();

    // ── Method 1: From an existing Solid ──
    Solid sourceSolid = GetSolid(someElement);  // your helper
    if (sourceSolid != null)
    {
        DirectShape ds = DirectShape.CreateElement(doc,
            new ElementId(BuiltInCategory.OST_GenericModel));
        ds.SetShape(new GeometryObject[] { sourceSolid });
        ds.Name = "Clash Volume";
    }

    // ── Method 2: From TessellatedShapeBuilder (arbitrary mesh) ──
    var builder = new TessellatedShapeBuilder();
    builder.OpenConnectedFaceSet(true);  // true = closed solid

    // Add 4 triangular faces of a tetrahedron
    XYZ[] verts = {
        new XYZ(0,0,0), new XYZ(5,0,0),
        new XYZ(0,5,0), new XYZ(0,0,5)
    };
    builder.AddFace(new TessellatedFace(
        new List<XYZ> { verts[0], verts[1], verts[2] },
        ElementId.InvalidElementId));
    builder.AddFace(new TessellatedFace(
        new List<XYZ> { verts[0], verts[1], verts[3] },
        ElementId.InvalidElementId));
    builder.AddFace(new TessellatedFace(
        new List<XYZ> { verts[0], verts[2], verts[3] },
        ElementId.InvalidElementId));
    builder.AddFace(new TessellatedFace(
        new List<XYZ> { verts[1], verts[2], verts[3] },
        ElementId.InvalidElementId));

    builder.CloseConnectedFaceSet();
    builder.Build();
    TessellatedShapeBuilderResult result = builder.GetBuildResult();

    DirectShape ds2 = DirectShape.CreateElement(doc,
        new ElementId(BuiltInCategory.OST_GenericModel));
    ds2.SetShape(result.GetGeometricalObjects());
    ds2.Name = "Tetrahedron";

    tx.Commit();
}
ClassSketchPlane
Inherits: Element
Namespace: Autodesk.Revit.DB · Family documents only
A reference plane used as the base for sketching family geometry. Required for creating extrusions, blends, revolves and sweeps in family documents. Created from a Plane using SketchPlane.Create().
🟢 Factory + Extrusion Pattern
🟣public static SketchPlane Create(Document doc, Plane geometryPlane)
Creates a SketchPlane from a geometric Plane. The SketchPlane defines the drawing plane for family geometry (extrusions, sweeps, revolves).
// Example — Create extrusion in family document
// Must be in a family document
if (!famDoc.IsFamilyDocument) return;

using (var tx = new Transaction(famDoc, "Create Extrusion"))
{
    tx.Start();

    // 1. Create sketch plane (XY plane at Z=0)
    Plane       plane      = Plane.CreateByNormalAndOrigin(XYZ.BasisZ, XYZ.Zero);
    SketchPlane sketchPln  = SketchPlane.Create(famDoc, plane);

    // 2. Create profile as CurveArrArray
    double wFt = UnitUtils.ConvertToInternalUnits(600, UnitTypeId.Millimeters);
    double hFt = UnitUtils.ConvertToInternalUnits(400, UnitTypeId.Millimeters);

    var loop = new CurveArray();
    loop.Append(Line.CreateBound(new XYZ(0,0,0), new XYZ(wFt,0,0)));
    loop.Append(Line.CreateBound(new XYZ(wFt,0,0), new XYZ(wFt,hFt,0)));
    loop.Append(Line.CreateBound(new XYZ(wFt,hFt,0), new XYZ(0,hFt,0)));
    loop.Append(Line.CreateBound(new XYZ(0,hFt,0), new XYZ(0,0,0)));

    var profile = new CurveArrArray();
    profile.Append(loop);

    // 3. Create the extrusion
    double depth = UnitUtils.ConvertToInternalUnits(2100, UnitTypeId.Millimeters);
    Extrusion extr = famDoc.FamilyCreate.NewExtrusion(
        true,       // isSolid
        profile,
        sketchPln,
        depth);

    // 4. Assign material
    ElementId matId = new FilteredElementCollector(famDoc)
        .OfClass(typeof(Material)).Cast<Material>()
        .First(m => m.Name.Contains("Wood")).Id;
    extr.get_Parameter(BuiltInParameter.MATERIAL_ID_PARAM).Set(matId);

    tx.Commit();
}

// Print & Export

Print · PDF · DWG · IFC · Image Export

Export sheets and views to all standard formats programmatically.

ClassPrintManager
Inherits: APIObject
Namespace: Autodesk.Revit.DB
Manages print operations for a document. Allows setting the printer, paper size, print range (all sheets, selection, or current view), and submitting print jobs. Access via doc.PrintManager.
🟢 Key Usage Pattern
// Example — Print all sheets to a printer
PrintManager pm = doc.PrintManager;

// Set printer name
pm.SelectNewPrintDriver("Microsoft Print to PDF");

// Configure print setup
pm.PrintRange = PrintRange.Select;
pm.CombinedFile = false;   // one file per sheet

// Build view set of all sheets
ViewSet viewSet = pm.ViewSheetSetting.CurrentViewSheetSet.Views;
var sheets = new FilteredElementCollector(doc)
    .OfClass(typeof(ViewSheet))
    .Cast<ViewSheet>()
    .OrderBy(s => s.SheetNumber)
    .ToList();

using (var tx = new Transaction(doc, "Print"))
{
    tx.Start();
    var viewSheetSet = pm.ViewSheetSetting.CurrentViewSheetSet;
    viewSheetSet.Views = new ViewSet();
    foreach (var s in sheets) viewSheetSet.Views.Insert(s);
    pm.ViewSheetSetting.SaveAs("All Sheets");
    tx.Commit();
}

// Submit print job (shows print dialog unless InSession=true)
pm.SubmitPrint();
System.Diagnostics.Debug.WriteLine($"Sent {sheets.Count} sheets to printer");
ClassPDFExportOptions
Inherits: ExportOptions
Namespace: Autodesk.Revit.DB
Options for exporting views or sheets to PDF format. Introduced in Revit 2022 as a native PDF export (replacing the virtual printer workaround). Supports individual files per sheet or combined multi-page PDF.
🟢 PDF Export Pattern
// Example — Export all sheets as individual PDFs
// Build list of sheet view IDs
var sheetIds = new FilteredElementCollector(doc)
    .OfClass(typeof(ViewSheet))
    .ToElementIds()
    .ToList();

var opts = new PDFExportOptions
{
    FileName                    = "Project Drawings",
    Combine                     = false,   // false = one PDF per sheet
    ExportQuality               = PDFExportQualityType.DPI300,
    ColorDepth                  = ColorDepthType.Color,
    RasterQuality               = RasterQualityType.High,
    HideScopeBoxes              = true,
    HideReferencePlanes         = true,
    HideUnreferencedViewTags    = true
};

// Set naming rule: sheet number + sheet name
opts.SetNamingRule("{Sheet Number}-{Sheet Name}");

// Export
bool success = doc.Export(
    @"C:\Output\PDFs",   // folder
    sheetIds,
    opts);

System.Diagnostics.Debug.WriteLine(
    success ? $"Exported {sheetIds.Count} PDFs" : "Export failed");

// Combined single PDF
opts.Combine  = true;
opts.FileName = "All_Sheets_Combined";
doc.Export(@"C:\Output", sheetIds, opts);

// Worksharing

WorksharingUtils · Workset · WorksetTable

Query workset ownership, move elements between worksets, check out elements.

ClassWorksharingUtils
Inherits: Static utility
Namespace: Autodesk.Revit.DB
Static utility class for workshared document operations — checking out elements, querying ownership status, and getting checkout information. Only meaningful for workshared (central file) models.
🟢 Key Static Methods
🟢public static CheckoutStatus GetCheckoutStatus(Document doc, ElementId elementId)
Returns the checkout/ownership status of the specified element. Check this before attempting to modify elements in a workshared model.
Returns: CheckoutStatus — OwnedByCurrentUser, OwnedByOtherUser, NotOwned, or Unknown.
🟢public static ICollection<ElementId> CheckoutElements(Document doc, ICollection<ElementId> elementIds)
Attempts to check out (borrow) the specified elements. Returns the IDs of elements that were successfully checked out. Elements owned by other users are silently skipped.
Returns: ICollection<ElementId> — IDs of successfully checked-out elements.
🟢public static WorksharingTooltipInfo GetWorksharingTooltipInfo(Document doc, ElementId elementId)
Returns detailed ownership information including owner name, last changed by, and last changed on date. Use for building ownership audit reports.
// Example — Worksharing audit
if (!doc.IsWorkshared) return;

var walls = new FilteredElementCollector(doc)
    .OfClass(typeof(Wall)).ToElementIds();

var ownedByOthers = new List<ElementId>();
var ownedByMe     = new List<ElementId>();

foreach (var id in walls)
{
    CheckoutStatus status =
        WorksharingUtils.GetCheckoutStatus(doc, id);

    switch (status)
    {
        case CheckoutStatus.OwnedByCurrentUser:
            ownedByMe.Add(id); break;
        case CheckoutStatus.OwnedByOtherUser:
            ownedByOthers.Add(id);
            // Get owner name
            var info = WorksharingUtils
                .GetWorksharingTooltipInfo(doc, id);
            System.Diagnostics.Debug.WriteLine(
                $"Owned by: {info.Owner} | Last changed: {info.LastChangedBy}");
            break;
    }
}

// Try to check out un-owned elements
var unowned = walls.Where(id =>
    WorksharingUtils.GetCheckoutStatus(doc, id)
        == CheckoutStatus.NotOwned).ToList();

var checkedOut = WorksharingUtils.CheckoutElements(doc, unowned);
System.Diagnostics.Debug.WriteLine(
    $"Checked out {checkedOut.Count} of {unowned.Count} requested");
ClassWorkset
Inherits: APIObject
Namespace: Autodesk.Revit.DB
Represents a single workset in a workshared model. Provides the workset name, ID, kind, and whether it's currently open. Worksets are managed via WorksetTable obtained from doc.GetWorksetTable().
🟠 Properties + WorksetTable Usage
🟠public WorksetId Id { get; }
Gets the workset's unique identifier. Use this ID to assign elements to a workset via ELEM_PARTITION_PARAM.
🟠public string Name { get; }
Gets the display name of the workset.
🟠public WorksetKind Kind { get; }
Gets the workset kind: UserCreated, FamilyWorkset, ViewWorkset, or StandardWorkset.
// Example — Workset operations
if (!doc.IsWorkshared) return;

WorksetTable wt = doc.GetWorksetTable();

// List all user-created worksets
IList<Workset> worksets =
    WorksetTable.GetWorksets(doc, WorksetKind.UserCreated);

foreach (var ws in worksets)
{
    // Count elements on this workset
    int count = new FilteredElementCollector(doc)
        .WherePasses(new ElementWorksetFilter(ws.Id))
        .GetElementCount();

    System.Diagnostics.Debug.WriteLine(
        $"Workset: {ws.Name} | Elements: {count} | Kind: {ws.Kind}");
}

// Move a wall to a specific workset
Workset targetWs = worksets.First(w => w.Name == "Architecture");

using (var tx = new Transaction(doc, "Move to Workset"))
{
    tx.Start();
    wall.get_Parameter(BuiltInParameter.ELEM_PARTITION_PARAM)
        .Set(targetWs.Id.IntegerValue);
    tx.Commit();
}

// Create a new workset
using (var tx = new Transaction(doc, "New Workset"))
{
    tx.Start();
    Workset.Create(doc, "MEP Systems");
    tx.Commit();
}

ClassDimension
Inherits: Element

Represents a dimension annotation in a view. Dimensions are created using doc.Create.NewDimension() with a ReferenceArray and a dimension line. They can be single-segment or multi-segment. Dimension values are read-only (derived from geometry) but text can be overridden.

// ── REFERENCE TYPES SUMMARY ──
// Wall face:      face.Reference (from get_Geometry with ComputeReferences=true)
// Grid:           new Reference(grid)
// Curve endpoint: curve.GetEndPointReference(0 or 1)
// FamilyInstance: fi.GetReferences(FamilyInstanceReferenceType.Left)

GeometryElement geom = wall.get_Geometry(opts);
Reference face1 = null, face2 = null;

foreach (GeometryObject obj in geom)
{
    if (obj is Solid s)
        foreach (Face f in s.Faces)
        {
            XYZ n = f.ComputeNormal(new UV(.5,.5));
            if      (n.X >  0.9) face1 = f.Reference;  // exterior face
            else if (n.X < -0.9) face2 = f.Reference;  // interior face
        }
}

ReferenceArray refs = new ReferenceArray();
refs.Append(face1);
refs.Append(face2);

XYZ p1     = (wall.Location as LocationCurve).Curve.GetEndPoint(0);
Line dimLn = Line.CreateBound(
    p1 + new XYZ(0, 3, 0),
    p1 + new XYZ(10, 3, 0));

{
    tx.Start();
    Dimension dim = doc.Create.NewDimension(doc.ActiveView, dimLn, refs);
    tx.Commit();

    // Read value
    if (dim.Value.HasValue)
    {
        double mm = dim.Value.Value * 304.8;
        System.Diagnostics.Debug.WriteLine($"Wall thickness: {mm:F0}mm");
    }
}

// Multi-segment: read each segment
Dimension multiDim = ...;
foreach (DimensionSegment seg in multiDim.Segments)
{
